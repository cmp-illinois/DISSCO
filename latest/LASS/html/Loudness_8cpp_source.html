43void Loudness::calculate(Sound& snd, m_rate_type rate)
44{
45 // count the number of Partials:
46 int numPartials = snd.size();
47
48 // Create a linear interpolator for each partial.
49 // this will become that partial's loudness scaling factor
50 vector<LinearInterpolator> scalingFactors(numPartials);
51
52 // Set each partial's Frequency
53 // Dynamic Variable to the proper duration and sampling rate:
54 m_time_type duration = snd.getParam(DURATION);
55 // Also, Collect a frequency iterator from each partial
56 vector< Iterator<m_value_type> > freqIter;
57 // finally, grab the maximum waveshape value from each partial
58 vector<m_value_type> maxWaveShape;
59 // go:
60 for (int i=0; i<numPartials; i++)
61 {
62 Partial& part = snd.get(i);
63 part.getParam(FREQUENCY).setDuration(duration);
64 part.getParam(FREQUENCY).setSamplingRate(rate);
65 freqIter.push_back( part.getParam(FREQUENCY).valueIterator() );
66 maxWaveShape.push_back( part.getParam(WAVE_SHAPE).getMaxValue() );
67 }
68
69 // create a vector of critical band objects (one for each valid band)
70 vector<CriticalBand> CBands;
71 for (int i=0; i<NUM_BANDS; i++)
72 CBands.push_back( CriticalBand(i) );
73
74 // calculate the number of samples to do loudness on:
75 m_sample_count_type numSamples = (m_sample_count_type) ((m_time_type)rate * (m_time_type)duration);
76
77 // iterate over time:
78 for (int s=0; s<numSamples; s++)
79 {
80 // calculate a relative time [0,1]
81 m_time_type relativeTime = float(s) / float(numSamples);
82
83 // the loudestPartial:
84 m_value_type maxAmp = 0.0;
85
86 // for each partial
87 for (int p=0; p<numPartials; p++)
88 {
89 // get the values:
90 m_value_type freq = freqIter[p].next();
91 m_value_type waveShape = maxWaveShape[p];
92
93 // scale the amplitude by the relative amplitude:
94 m_value_type thisAmp = waveShape * snd.get(p).getParam(RELATIVE_AMPLITUDE);
95
96 // is this the loudest partial?
97 if (thisAmp > maxAmp) maxAmp = thisAmp;
98
99 // create a snapshot for this moment in time:
100 PartialSnapshot snap(p, freq, waveShape);
101
102 // add this snapshot to the appropriate critical band:
103 CBands[criticalBandIndex(freq)].partials_.push_back(snap);
104 }
105
106 // calculate the band gamma for each band:
107 vector<m_value_type> bandGamma(NUM_BANDS);
108 for (int i=0; i<NUM_BANDS; i++)
109 bandGamma[i] = CBands[i].getBandGamma(maxAmp);
110
111 // find the maximum band gamma:
112 m_value_type maxGamma = 0;
113 for (int i=0; i<NUM_BANDS; i++)
114 if (bandGamma[i] > bandGamma[(int)maxGamma]) bandGamma[(int)maxGamma] = bandGamma[i];
115
116 // calculate the numerator:
117 m_value_type gammaTotal = 0.0;
118 for (int i=0; i<NUM_BANDS; i++)
119 if (i != maxGamma) gammaTotal += bandGamma[i] * BANDS[i][F_FACTOR];
120 m_value_type numerator = snd.getParam(LOUDNESS) / (bandGamma[(int)maxGamma] + gammaTotal);
121
122 // for each band:
123 for (int b=0; b<NUM_BANDS; b++)
124 {
125 int size = CBands[b].partials_.size();
126 // for each partial
127 for (int p=0; p<size; p++)
128 {
129 //calculate the scaling factor
130 m_value_type scaleFactor = CBands[b].partials_[p].getScalingFactor(
131 b, maxAmp, numerator);
132 //grab the id of this partial:
133 int partial_id = CBands[b].partials_[p].ID_;
134
135 // scale the factor by the relative amplitude again:
136 scaleFactor *= snd.get(partial_id).getParam(RELATIVE_AMPLITUDE);
137
138 //add the scaling factor for this moment in time.
139 scalingFactors[partial_id].addEntry(relativeTime, scaleFactor);
140 }
141 }
142
143
144 // clear the critical bands:
145 for (int i=0; i<NUM_BANDS; i++)
146 CBands[i].partials_.clear();
147
148
149 } // end iterating over time
150
151 // set every partial's scaling factors:
152 for (int p=0; p<numPartials; p++)
153 snd.get(p).setParam(LOUDNESS_SCALAR, scalingFactors[p]);
154}

159int Loudness::criticalBandIndex(m_value_type freq)
160{
161 // border conditions:
162 if (freq < BANDS[0][LOWER_BOUND])
163 {
164 return 0;
165 cout << "WARNING: Frequency (" << freq
166 << ") outside of any critical bands." << endl;
167 }
168 if (freq > BANDS[NUM_BANDS-1][UPPER_BOUND])
169 {
170 return NUM_BANDS-1;
171 cout << "WARNING: Frequency (" << freq
172 << ") outside of any critical bands." << endl;
173 }
174
175 // common case:
176 for (int i=0; i<NUM_BANDS; i++)
177 {
178 if (freq < BANDS[i][UPPER_BOUND])
179 return i;
180 }
181
182 cout << "ERROR: didn't find a critical band for this frequency." << freq << endl;
183 return NUM_BANDS-1;
184}

197m_value_type Loudness::CriticalBand::getBandGamma(m_value_type maxAmp)
198{
199 m_value_type bandGamma = 0.0;
200
201 for(int p = 0; p < (int)partials_.size(); p++)
202 {
203 bandGamma += pow(
204 (double)(partials_[p].amp_ / maxAmp),
205 (double)(1.0 / log10(2.0) / BANDS[ID_][SLOPE]) );
206 }
207
208 bandGamma = pow(
209 (double)bandGamma,
210 (double)(log10(2.0) / BANDS[ID_][SLOPE]) );
211
212 return bandGamma;
213}

224//----------------------------------------------------------------------------//
225m_value_type Loudness::PartialSnapshot::getScalingFactor(
226 int bandID, m_value_type maxAmp, m_value_type numerator)
227{
228 // [eq 2.37]
229 m_value_type Ls = (amp_ / maxAmp) * numerator;
230
231 // [eq 2.4]
232 m_value_type Lp = log(Ls) / log(2.0) * 10.0 + 40.0;
233
234 // [eq 2.13]
235 m_value_type L = BANDS[bandID][OFFSET] + ( BANDS[bandID][SLOPE] * Lp );
236
237 // [eq 2.5]
238 m_value_type A = pow(10.0, (-1.0 * ( (120.0 - L) / 20.0 ) ) );
239
240
241 return (A / amp_);
242}
