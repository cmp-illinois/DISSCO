123/*************************************************************************/
124// This is a very useful function to essentially find the y-value of the
125// x-value (time-value) passed in as an argument. If the point asked for
126// is at the start of a segment, this is an easy function. However, if it
127// is not, the value must be interpolated, either linearly or exponentially,
128// because these are the two path types currently defined.
129//
130// The exponential interpolation was derived by Hans Kaper one day. It
131// is correct. Basically, the derivation starts with the exponential
132// equation y(t) = y1*e^(L*(t-t1)) and through simplification and such,
133// gets to the following equation:
134//
135// 1 + ((y(t) - y1)/y1) = (1 + ((y2 - y1)/y1)) ^ ((t - t1)/(t2 - t1))
136//
137// Then, the following simplifications are done:
138//
139// X = (y(t) - y1)/y1
140// dy = (y2 - y1)/y1
141// dt = (t - t1)/(t2 - t1)
142//
143// So, the equation becomes:
144//
145// 1 + X = (1 + dy) ^ dt
146//
147// which is what is used in this function. This shows a very important
148// detail. This gets the value relative to this exponential function,
149// not an absolute value.
150/*************************************************************************/
151
152m_value_type& ExponentialInterpolatorIterator::next()
153{
154 if( queue_.empty() ) return value_;
155
156 Entry& e = queue_.front();
157 if( ! stepsLeft_ ) // we're at the beginning of an entry, so just return
158 // that value and set up the steps for the next
159 {
160 stepsLeft_ = e.steps_;
161 value_ = e.v_from_;
162 }
163 if (stepsLeft_ > 0)
164 {
165 stepsLeft_--;
166 m_value_type y1 = e.v_from_;
167 m_value_type y2 = e.v_to_;
168 // m_value_type dy;
169
170 if( y1 == 0 )
171 y1 = .0001;
172 if( y2 == 0 )
173 y2 = .0001;
174
175 m_value_type alpha = 3;
176 if (y1 > y2)
177 {
178 alpha = -alpha;
179 }
180
181 // unused parameter dy
182 // dy = (y2 - y1) / y1;
183 m_time_type t1 = e.t_from_;
184 m_time_type t2 = e.t_to_;
185 m_time_type t = (((t2 - t1) / e.steps_) * (e.steps_ - stepsLeft_)) + t1;
186 m_value_type I = (t - t1) / (t2 - t1);
187 m_value_type base = 2.718282;
188 value_ = y1 + (y2 - y1) * ((1 - pow (base, (I * alpha))) / (1 - pow (base, alpha)));
189 }
190
191 if( ! stepsLeft_ )
192 {
193 queue_.pop_front();
194 if( queue_.empty() ) return value_;
195 e = queue_.front();
196 stepsLeft_ = e.steps_;
197 value_ = e.v_from_;
198 }
199
200 return value_;
201}

213/*************************************************************************/
214// The ith cubic polynomial u_i(x) which connects (xi,yi) with slope
215// mi to (x_i+1,y_i+1) with slope m_i+1 as x ranges from xi to
216// x_i+1 is:
217//
218// u_i(x) = ai+bi(x-xi)+ci(x-xi)^2+di(x-xi)^3, where
219// ai = yi,
220// bi = mi,
221// ci = (3(y_i+1-yi)/(x_i+1-xi) - 2mi - m_i+1) / (x_i+1 - xi)
222// di = (mi + m_i+1 - 2(y_i+1 - yi)/(x_i+1 - xi)) / (x_i+1 - xi)^2
223//
224// In this function, we use:
225// mi = (y_i+1-y_i-1)/(x_i+1-x_i-1) for i=2,...,n-1 which is just the
226// average of the slopes surrounding a point, and for the ends:
227// m1 = (y2-y1)/(x2-x1), mn = (yn-y_n-1)/(xn-x_n-1)
228//
229// This is achieved by taking four (time,value) points and using them
230// to make the above calculations: two points specifying the range,
231// and a point before and one after.
232/*************************************************************************/
233m_value_type& CubicSplineInterpolatorIterator::next()
234{
235
236 if( queue_.empty() ) return value_;
237
238 Entry e = queue_.front();
239 // the line below is substituted by the line above --Ming-ching
240 //Entry& e = queue_.front();
241 queue_.pop_front();
242 int chk = 0;
243 if( queue_.empty() ) { chk = 1; }
244 else { // get the point ahead of the range since there are more
245 // entries in the queue
246 Entry& e2 = queue_.front();
247 x3 = e2.t_to_;
248 y3 = e2.v_to_;
249 }
250 queue_.push_front(e);
251
252 // we need this in the larger scope
253 m_time_type x1 = 0.0;
254 m_value_type y1 = 0.0;
255
256 if( ! stepsLeft_ ) // we're at the beginning of an entry, so just return
257 // that value and set up the steps for the next
258 {
259 stepsLeft_ = e.steps_;
260 value_ = e.v_from_;
261 }
262 else if (stepsLeft_ > 0)
263 {
264 stepsLeft_--;
265 x1 = e.t_from_;
266 m_time_type x2 = e.t_to_;
267 m_time_type x = (( ( x2 - x1 ) / e.steps_ ) * ( e.steps_ - stepsLeft_ )) + x1;
268 y1 = e.v_from_;
269 m_value_type y2 = e.v_to_;
270
271 // since we can't lookahead, just give the next points the end of
272 // the range's values
273 if( chk ) { x3 = x2; y3 = y2; }
274
275 m_value_type l_slp = (y2-y1)/(m_value_type)(x2-x1);
276
277 m_value_type m1;
278 if( x0 == -1 && y0 == -1 ) // front of queue
279 m1 = l_slp;
280 else
281 m1 = (y2-y0)/(m_value_type)(x2-x0);
282
283 m_value_type m2 = (y3-y1)/(x3-x1);
284
285 // our coefficients for the polynomial
286 m_value_type a = y1;
287 m_value_type b = m1;
288 m_value_type c = ((3*l_slp)-(2*m1)-m2)/(x2-x1);
289 m_value_type d = (m1+m2-(2*l_slp))/(pow((x2-x1),2));
290
291 m_value_type x_dif = x - x1;
292
293 // the polynomial
294 m_value_type ux = a + (b*x_dif) + (c*(pow(x_dif,2))) + (d*(pow(x_dif,3)));
295
296 value_ = ux;
297 }
298
299 if( ! stepsLeft_ ) {
300
301 // set the last point to the beginning of the current point
302 // as we move ahead
303 x0 = x1;
304 y0 = y1;
305 queue_.pop_front();
306 if( queue_.empty() ) return value_;
307 e = queue_.front();
308 stepsLeft_ = e.steps_;
309 value_ = e.v_from_;
310 }
311
312 return value_;
313}
