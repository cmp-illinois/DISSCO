43AllPassFilter::AllPassFilter(float gain, long delay)
44{
45 // set parameters
46 g = gain;
47 g_sqrd = g*g;
48 D = delay;
49
50
51 // initialize x_hist queue and y_hist queue
52 x_hist = new Filter::hist_queue<m_sample_type>(D);
53 for(long i=0; i<D; i++)
54 x_hist->enqueue(0.0);
55
56 y_hist = new Filter::hist_queue<m_sample_type>(D);
57 for(long j=0; j<D; j++)
58 y_hist->enqueue(0.0);
59}

62AllPassFilter::~AllPassFilter()

64 delete y_hist;
65 delete x_hist;
66}

70m_sample_type AllPassFilter::do_filter(m_sample_type x_t)
71{
72 m_sample_type y_t;
73
74 y_t=-g*x_t + (1-g_sqrd)*(x_hist->dequeue() + g*y_hist->dequeue());
75 x_hist->enqueue(x_t);
76 y_hist->enqueue(y_t);
77 return y_t;
78}
