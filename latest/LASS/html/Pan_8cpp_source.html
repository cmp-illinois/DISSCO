54MultiTrack* Pan::spatialize_Track(Track& t, int numTracks)
55{
56 m_sample_count_type sampleCount = t.getWave().getSampleCount();
57 m_rate_type samplingRate = t.getWave().getSamplingRate();
58
59 // create an empty multi Track object:
60 MultiTrack* mt = new MultiTrack(numTracks, sampleCount, samplingRate);
61
62 // tell panVar_ this information.
63 panVar_->setSamplingRate(samplingRate);
64 panVar_->setDuration( float(sampleCount) / float(samplingRate));
65
66 // get references to the input
67 SoundSample& inWave = t.getWave();
68 bool doAmp = t.hasAmp();
69 SoundSample* inAmp = doAmp ? &t.getAmp() : 0;
70
71
72 // for each channel:
73 for (int c=0; c<numTracks; c++)
74 {
75 // get an iterator for the pan variable:
76 Iterator<m_value_type> panIter = panVar_->valueIterator();
77
78 // get references for this channel.
79 SoundSample& thisWave = mt->get(c)->getWave();
80 SoundSample* thisAmp = doAmp ? &mt->get(c)->getAmp() : 0;
81
82 // create a position value for this channel:
83 m_value_type pos = float(c) / float(numTracks-1);
84
85 // Iterate over each sample
86 m_value_type scale;
87 for (m_sample_count_type i=0; i<sampleCount; i++)
88 {
89 // calculate the scaling factor:
90 scale = 1.0 - fabs(pos - panIter.next());
91 if (scale > 1.0) scale = 1.0;
92 if (scale < 0.0) scale = 0.0;
93
94 thisWave[i] = scale * inWave[i];
95 if (doAmp) (*thisAmp)[i] = scale * (*inAmp)[i];
96 }
97 }
98
99 // Return the multitrack object
100 return mt;
101}
