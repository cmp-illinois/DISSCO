62MultiTrack* Partial::render(int numChannels,
63 m_sample_count_type sampleCount,
64 m_time_type duration,
65 m_rate_type samplingRate)
66{
67 // calculate the number of samples needed:
68 m_sample_count_type numSamplesToRender = (m_sample_count_type) (duration * (m_time_type)samplingRate);
69
70 //unused parameter numSamplesTotal
71 // m_sample_count_type numSamplesTotal;
72 // if(reverbObj != NULL)
73 // numSamplesTotal = (m_sample_count_type) ((duration + reverbObj->getDecay()) * (m_time_type)samplingRate);
74 // else
75
76 // numSamplesTotal = (m_sample_count_type) (duration * (m_time_type)samplingRate);
77 // create two SoundSample objects to write to:
78 // one for the actuial sound:
79 // SoundSample* waveSample = new SoundSample(numSamplesTotal, samplingRate);
80 SoundSample* waveSample = new SoundSample(sampleCount, samplingRate);
81 // and one to store amplitude data (only when a downstream consumer, i.e. an
82 // amplitude-based clipping mode, actually needs it -- otherwise this would
83 // double the rendering memory for nothing):
84 // SoundSample* ampSample = new SoundSample(numSamplesTotal, samplingRate);
85 SoundSample* ampSample = Track::amplitudeTrackingEnabled()
86 ? new SoundSample(sampleCount, samplingRate)
87 : 0;
88
89 // tell each dynamic variable what the DURATION will be:
90 getParam(FREQUENCY).setDuration(duration);
91 getParam(WAVE_SHAPE).setDuration(duration);
92 getParam(TREMOLO_AMP).setDuration(duration);
93 getParam(TREMOLO_RATE).setDuration(duration);
94 getParam(VIBRATO_AMP).setDuration(duration);
95 getParam(VIBRATO_RATE).setDuration(duration);
96 getParam(PHASE).setDuration(duration);
97 getParam(LOUDNESS_SCALAR).setDuration(duration);
98 //getParam(FREQUENCY_DEVIATION).setDuration(duration);
99 //getParam(GLISSANDO_ENV).setDuration(duration);
100 getParam(FREQ_ENV).setDuration(duration);
101 getParam(DETUNING_ENV).setDuration(duration);
102 getParam(AMPTRANS_WIDTH).setDuration(duration);
103 getParam(FREQTRANS_WIDTH).setDuration(duration);
104
105 // tell each dynamic variable what the SAMPLING_RATE will be:
106 getParam(FREQUENCY).setSamplingRate(samplingRate);
107 getParam(WAVE_SHAPE).setSamplingRate(samplingRate);
108 getParam(TREMOLO_AMP).setSamplingRate(samplingRate);
109 getParam(TREMOLO_RATE).setSamplingRate(samplingRate);
110 getParam(VIBRATO_AMP).setSamplingRate(samplingRate);
111 getParam(VIBRATO_RATE).setSamplingRate(samplingRate);
112 getParam(PHASE).setSamplingRate(samplingRate);
113 getParam(LOUDNESS_SCALAR).setSamplingRate(samplingRate);
114 //getParam(FREQUENCY_DEVIATION).setSamplingRate(samplingRate);
115 //getParam(GLISSANDO_ENV).setSamplingRate(samplingRate);
116 getParam(FREQ_ENV).setSamplingRate(samplingRate);
117 getParam(DETUNING_ENV).setSamplingRate(samplingRate);
118 getParam(AMPTRANS_WIDTH).setSamplingRate(samplingRate);
119 getParam(FREQTRANS_WIDTH).setSamplingRate(samplingRate);
120
121 // randomize the frequency deviation:
122 //
123 // calculate the variance:
124 // grab the frequency at the start of the partial:
125 //m_value_type base_freq = getParam(FREQUENCY).valueIterator().next();
126 //m_value_type freq_variance=(base_freq/(getParam(PARTIAL_NUM)+1.0))/2.0;
127 //
128 // get a random nummber between -1 and 1:
129 //m_value_type rand = (((double)random())/((double)RAND_MAX)*2.0)-1.0;
130 //
131 // vary the freq_dev by this value:
132 // make a copy of the frequency deviation dynamic variable
133 //DynamicVariable* freq_dev = getParam(FREQUENCY_DEVIATION).clone();
134 // scale the copy:
135 //freq_dev->scale(rand * freq_variance);
136 //
137 // make a copy of the glissando dynamic variable
138
139//cout << "Partial:: render - frequency before detune:" << getParam(FREQ_ENV).getMaxValue() << endl;
140 DynamicVariable* frequency_env = getParam(FREQUENCY).clone();
141 frequency_env->setDuration(duration);
142
143 //DynamicVariable* gliss_env = getParam(GLISSANDO_ENV).clone();
144 DynamicVariable* freq_env = getParam(FREQ_ENV).clone();
145 DynamicVariable* detuning_env = getParam(DETUNING_ENV).clone();
146 DynamicVariable* amptrans_amp_env = getParam(AMPTRANS_AMP_ENV).clone();
147 DynamicVariable* amptrans_rate_env = getParam(AMPTRANS_RATE_ENV).clone();
148 DynamicVariable* freqtrans_amp_env = getParam(FREQTRANS_AMP_ENV).clone();
149 DynamicVariable* freqtrans_rate_env = getParam(FREQTRANS_RATE_ENV).clone();
150 //gliss_env->setDuration(duration);
151 freq_env->setDuration(duration);
152 detuning_env->setDuration(duration);
153 amptrans_amp_env->setDuration(duration);
154 amptrans_rate_env->setDuration(duration);
155 freqtrans_amp_env->setDuration(duration);
156 freqtrans_rate_env->setDuration(duration);
157
158 // grab iterators for each DynamicVariable:
159 typedef Iterator<m_value_type> ValIter;
160
161 // ValIter freq_it = getParam(FREQUENCY).valueIterator();
162 //more messing around
163 ValIter frequency_it = frequency_env->valueIterator();
164
165 ValIter wave_shape_it = getParam(WAVE_SHAPE).valueIterator();
166 ValIter tremolo_amp_it = getParam(TREMOLO_AMP).valueIterator();
167 ValIter tremolo_rate_it = getParam(TREMOLO_RATE).valueIterator();
168 ValIter vibrato_amp_it = getParam(VIBRATO_AMP).valueIterator();
169 ValIter vibrato_rate_it = getParam(VIBRATO_RATE).valueIterator();
170 ValIter phase_it = getParam(PHASE).valueIterator();
171 ValIter loudnes_scalar_it = getParam(LOUDNESS_SCALAR).valueIterator();
172 ValIter amptrans_width_it = getParam(AMPTRANS_WIDTH).valueIterator();
173 ValIter freqtrans_width_it = getParam(FREQTRANS_WIDTH).valueIterator();
174 //ValIter freq_deviation_it = freq_dev->valueIterator();
175 //ValIter gliss_it = gliss_env->valueIterator();
176 ValIter freq_it = freq_env->valueIterator();
177 ValIter detuning_it = detuning_env->valueIterator();
178 ValIter amptrans_amp_it = amptrans_amp_env->valueIterator();
179 ValIter amptrans_rate_it = amptrans_rate_env->valueIterator();
180 ValIter freqtrans_amp_it = freqtrans_amp_env->valueIterator();
181 ValIter freqtrans_rate_it = freqtrans_rate_env->valueIterator();
182
183 // Phase values start off at zero for now:
184 m_value_type tremolo_phase = 0.0;
185 m_value_type vibrato_phase = 0.0;
186 m_value_type freq_phase = 0.0;
187
188 // values used in the loop:
189 m_value_type tremolo;
190 m_value_type amplitude;
191 m_value_type vibrato;
192 m_value_type frequency;
193 m_value_type phase;
194 m_value_type sample;
195 m_value_type amptransient;
196 m_value_type freqtransient;
197 m_value_type amptrans_width = 0.0;
198 m_value_type freqtrans_width = 0.0;
199 m_time_type amptransprob;
200 m_time_type freqtransprob;
201
202 srand(time(0));
203
204 //flags to tell if we are in a transient
205 int amptransflag = 0;
206 int freqtransflag = 0;
207
208 //placemarkers once in a transient
209 int amptransplace = 0;
210 int freqtransplace = 0;
211
212 //counters to see if we should check for a transient
213 int amptranscheck = (int)amptrans_width;
214 int freqtranscheck = (int)freqtrans_width;
215 float random;
216
217 m_value_type freqmod = 0.0, trans_freqmod = 0;
218 m_value_type amplifier = 0.0, trans_amplifier = 0;
219 // loop over every sample:
220 for (m_sample_count_type s=0; s < numSamplesToRender; s++)
221 {
222
223
224 // AMPLITUDE:
225
226 //advance transient related iterators
227 amptrans_width = amptrans_width_it.next();
228 trans_amplifier = amptrans_amp_it.next();
229 amptransprob = amptrans_rate_it.next();
230
231 // grab the tremolo value
232 tremolo = tremolo_amp_it.next() * sin(2.0 * M_PI * tremolo_phase);
233 // increment the tremolo phase
234 tremolo_phase = pmod( tremolo_phase + (tremolo_rate_it.next() / samplingRate) );
235
236 //once counter reaches 0, check for transient
237 if(amptranscheck <= 0 && s+amptrans_width < numSamplesToRender)
238 {
239 random = ((float)std::rand()/(float)RAND_MAX);
240 if(random <= amptransprob)
241 {
242 trans_amplifier *= (float)std::rand()/(float)RAND_MAX;
243 if(((float)std::rand()/(float)RAND_MAX) <= 0.5)
244 trans_amplifier *= -1;
245 amplifier = trans_amplifier;
246 amptransflag = 1;
247 //we are at beginning of transient
248 amptransplace = 0;
249 }
250 //reset check counter
251 amptranscheck = (int)amptrans_width;
252 }
253
254 // augment the amplitude value by the tremolo value, and the loudness scale:
255 if(amptransflag)
256 {
257 //the following decides where we are inside the transient
258 //used to scale the transient multiplier effect
259 if(amptransplace < amptrans_width/2)
260 amptransient = amplifier*amptransplace/(amptrans_width/2);
261 else
262 amptransient = amplifier*(amptrans_width - amptransplace)/(amptrans_width/2);
263 amptransplace++;
264 if(amptransplace >= amptrans_width)
265 amptransflag = 0;
266 }
267 else
268 amptransient = 0;
269
270 //decrease the check counter
271 amptranscheck--;
272
273 amplitude = loudnes_scalar_it.next() * wave_shape_it.next() * (1.0 + tremolo);
274
275 //apply transient modifier to amplitude
276 amplitude = amplitude + amptransient*amplitude;
277
278
279 // FREQUENCY:
280
281 //advance frequency transient related iterators
282 freqtrans_width = freqtrans_width_it.next();
283 trans_freqmod = freqtrans_amp_it.next();
284 freqtransprob = freqtrans_rate_it.next();
285
286 //if we should check, check
287 if(freqtranscheck <= 0 && s+freqtrans_width < numSamplesToRender)
288 {
289 random = ((float)std::rand()/(float)RAND_MAX);
290 if(random <= freqtransprob)
291 {
292 trans_freqmod *= (float)std::rand()/(float)RAND_MAX;
293 if(((float)std::rand()/(float)RAND_MAX) <= 0.5)
294 trans_freqmod *= -1;
295 freqmod = trans_freqmod;
296 freqtransflag = 1;
297 freqtransplace = 0;
298 }
299 freqtranscheck = (int)freqtrans_width;
300 }
301
302 //if we are inside a transient, make appropriate changes
303 if(freqtransflag)
304 {
305 if(freqtransplace < freqtrans_width/2)
306 freqtransient = freqmod*freqtransplace/(freqtrans_width/2);
307 else
308 freqtransient = freqmod*(freqtrans_width - freqtransplace)/(freqtrans_width/2);
309 freqtransplace++;
310 if(freqtransplace >= freqtrans_width)
311 freqtransflag = 0;
312 }
313 else
314 freqtransient = 0;
315 freqtranscheck--;
316
317 // perhaps the vibrato should be keyed to PartialNumber.
318
319 // grab the vibrato value
320 vibrato = vibrato_amp_it.next() * sin(2.0 * M_PI * vibrato_phase);
321 // increment the vibrato phase
322 vibrato_phase = pmod( vibrato_phase + (vibrato_rate_it.next() / samplingRate) );
323
324 // augment the frequency value by the freq-deviation and vibrato and detuning and freq_env and transients
325 frequency = frequency_it.next() * freq_it.next() * detuning_it.next();
326// frequency = frequency_it.next() * freq_it.next(); // * detuning_it.next()
327 //frequency += freq_deviation_it.next();
328 frequency *= 1.0 + vibrato;
329
330
331 //apply frequency transient adjustments
332 frequency += frequency*freqtransient;
333
334
335 // increment the frequency phase:
336 freq_phase = pmod (freq_phase + (frequency / samplingRate));
337
338 // calculate the actual phase, augmenting freq_phase with
339 // they dynamic variable phase (phase offset)
340 phase = freq_phase + phase_it.next();
341
342
343 // SAMPLE:
344
345 switch( (int) getParam(WAVE_TYPE))
346 {
347 case 1:
348 // random
349 sample = amplitude * 2 * ((((double) std::rand()) / ((double) RAND_MAX)) - 0.5);
350 break;
351
352 default:
353 // calculate the sample
354 sample = amplitude * ( sin(2.0 * M_PI * phase));
355 break;
356 }
357
358 // assign it:
359 (*waveSample)[s] = sample;
360 if (ampSample) (*ampSample)[s] = amplitude;
361 }
362
363 // create a TrackObject:
364 Track *_track = new Track(waveSample, ampSample);
365
366 // do the reverb, if necessary
367 if(reverbObj != NULL)
368 {
369 // do_reverb_Track() returns a freshly heap-allocated Track, so take
370 // ownership of it directly instead of deep-copying its full-length
371 // buffers into another new Track and then freeing the original.
372 Track &tmp = reverbObj->do_reverb_Track(*_track);
373 delete _track;
374 _track = &tmp;
375 }
376
377 /* ZIYUAN CHEN, July 2023: On the default behavior of spatialization (Partial side)
378 * If a non-placeholder subclass of Spatializer (Pan/MultiPan) is set in the sound,
379 * "spatializer_" in the partials will be placeholders that merely averages the
380 * sound evenly accross all tracks.
381 * But unlike Sound::render(), the placeholders are NOT IGNORED since they are essential
382 * for transforming Track to MultiTrack, whose channels uniformly hold scaled versions
383 * of the original Track.
384 * Compare Sound::render().
385 */
386 MultiTrack* returnTrack = spatializer_->spatialize_Track(*_track, numChannels);
387
388//cout << "Partial::render - frequency after detune:" << getParam(FREQ_ENV).getMaxValue() << endl;
389// cout<< "--------------------------------------------"<< endl;
390
391 // this section is added by Ming-ching on Dec.10 2012 to prevent memory leak
392 delete frequency_env;
393 delete freq_env;
394 delete amptrans_amp_env;
395 delete amptrans_rate_env;
396 delete freqtrans_amp_env;
397 delete freqtrans_rate_env;
398
399
400 return returnTrack;
401}

405{

407 num -= 1.0;

512void Partial::xml_read(XmlReader::xmltag* partialtag, DISSCO_HASHMAP<long, Reverb *>* reverbHash, DISSCO_HASHMAP<long, DynamicVariable *> *dvHash)
513{
514 if(strcmp("partial",partialtag->name))
515 {
516 printf("Not a partial tag! This is a %s tag.\n",partialtag->name);
517 return;
518 }
519
520 char *value;
521
522 if((value = partialtag->findChildParamValue("reverb_ptr", "id")) != 0)
523 {
524 long id=atoi(value);
525 Reverb * temp;
526 temp = (*reverbHash)[id];
527
528 if(temp != NULL)
529 {
530 use_reverb(temp);
531 }
532 }
533
534 if((value = partialtag->findChildParamValue(
535 "relative_amplitude", "value")) != 0)
536 setParam(RELATIVE_AMPLITUDE, atof(value));
537
538 if((value = partialtag->findChildParamValue("partial_num","value")) != 0)
539 setParam(PARTIAL_NUM, atoi(value));
540
541 // For sake of ease, I am going to forego searching for each item and instead
542 // iterate thru the list of child tags.
543
544 XmlReader::xmltag *dvtag; XmlReader::xmltagset *child=partialtag->children;
545
546 while(child)
547 {
548 dvtag=child->tag;
549
550 if(strcmp(dvtag->name,"frequency") == 0)
551 auxLoadParam(FREQUENCY,dvtag,dvHash);
552 else if(strcmp(dvtag->name,"wave_shape") == 0)
553 auxLoadParam(WAVE_SHAPE,dvtag,dvHash);
554 else if(strcmp(dvtag->name,"loudness_scalar") == 0)
555 auxLoadParam(LOUDNESS_SCALAR,dvtag,dvHash);
556 else if(strcmp(dvtag->name,"tremolo_amp") == 0)
557 auxLoadParam(TREMOLO_AMP,dvtag,dvHash);
558 else if(strcmp(dvtag->name,"tremolo_rate") == 0)
559 auxLoadParam(TREMOLO_RATE,dvtag,dvHash);
560 else if(strcmp(dvtag->name,"vibrato_amp") == 0)
561 auxLoadParam(VIBRATO_AMP,dvtag,dvHash);
562 else if(strcmp(dvtag->name,"vibrato_rate") == 0)
563 auxLoadParam(VIBRATO_RATE,dvtag,dvHash);
564 else if(strcmp(dvtag->name,"phase") == 0)
565 auxLoadParam(PHASE,dvtag,dvHash);
566 //else if(strcmp(dvtag->name,"frequency_deviation") == 0)
567 //auxLoadParam(FREQUENCY_DEVIATION,dvtag,dvHash);
568 //else if(strcmp(dvtag->name,"glissando_env") == 0)
569 //auxLoadParam(GLISSANDO_ENV,dvtag,dvHash);
570 else if(strcmp(dvtag->name,"freq_env") == 0)
571 auxLoadParam(FREQ_ENV,dvtag,dvHash);
572 else if(strcmp(dvtag->name,"detuning_env") == 0)
573 auxLoadParam(DETUNING_ENV,dvtag,dvHash);
574
575 child=child->next;
576 }
577}
