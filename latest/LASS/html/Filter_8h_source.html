194 ElemType dequeue(void)

199 back_idx = (back_idx + 1) % length;
200 return array[old_back_idx];

208 void enqueue(ElemType new_val)

220 ElemType index_from_oldest(long i)
221 {
222 long idx;
223
224 // we know idx%length, find the idx in the right range though
225 idx = back_idx+i;

227 idx-=length;
228
229 // return the index
