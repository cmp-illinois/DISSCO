64MultiTrack& MultiTrack::operator=(const MultiTrack& mt)
65{
66 if ( this != &mt) // beware of self assignment
67 {
68 // delete any tracks:
69 for (Track* track : tracks_)
70 delete track;
71
72 // clear this object
73 tracks_.clear();
74
75 // copy every track from mt to this object.
76 for (int i = 0; i < mt.size(); i++)
77 add(new Track(*mt.get(i)));
