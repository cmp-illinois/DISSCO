79DynamicVariable* DynamicVariable::create_dv_from_xml(XmlReader::xmltag *dvtag)
80{
81 // First look up the dv_type which is always a child tag
82
83 char *value=dvtag->findChildParamValue("dv_type","value");
84
85 DynamicVariable *dv=NULL;
86
87 // Make a new constant
88 if(!strcmp(value,"constant"))
89 {
90 Constant *constdv=new Constant;
91 constdv->xml_read(dvtag);
92 dv=constdv;
93 }
94 // Deprecated; ignoring.
95/*
96 else if(!strcmp(value,"dvs"))
97 {
98 value=dvtag->findChildParamValue("dv_ptr","value");
99 long id=atoi(value);
100 }
101*/
102 else if(!strcmp(value,"envelope"))
103 {
104 Envelope *envdv=new Envelope;
105 envdv->xml_read(dvtag);
106 dv=envdv;
107 }
108 else if(!strcmp(value,"interp"))
109 {
110 int interp_type = atoi(dvtag->findChildParamValue("interp_type","value"));
111 Interpolator *interp=NULL;
112 if(interp_type == EXPONENTIAL)
113 interp = new ExponentialInterpolator;
114 else if(interp_type == LINEAR)
115 interp = new LinearInterpolator;
116 else if(interp_type == CUBIC_SPLINE)
117 interp = new CubicSplineInterpolator;
118 interp->xml_read(dvtag);
119 dv=interp;
120 }
121
122 if(!dv)
123 {
124 printf("[%s:%d] Error creating DV: Unknown type %s\n",__FILE__,__LINE__,value);
125 return new Constant;
126 }
127
128 return dv;
129}
