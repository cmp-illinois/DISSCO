254void MultiPan::addEntryLocation(float t, float theta, float radius)
255{
256 typedef struct
257 {
258 float x, y, dist;
259 } Speaker;
260 Speaker **SpeakerList;
261 Speaker *curSpeaker;
262 float curTheta, curX, curY, total_dist;
263 int i;
264
265 if(useEnvDirectly == true)
266 {
267 cout << "Warning: called MultiPan::addEntry on a MultiPan object" << endl;
268 cout << " that is using ENVs directly" << endl;
269 }
270
271 // create an array of speakers
272 SpeakerList = new Speaker *[n_channels];
273 //cout << "Speaker Locations:" << endl;;
274 for(i=0;i<n_channels;i++)
275 {
276 curSpeaker = new Speaker();
277
278 curTheta = 2.0 * M_PI * (double)i / (double)n_channels;
279 curSpeaker->x = cos(curTheta);
280 curSpeaker->y = sin(curTheta);
281 //cout << "\tx = " << curSpeaker->x;
282 //cout << ", y=" << curSpeaker->y << endl;
283 SpeakerList[i] = curSpeaker;
284 }
285
286 // define a squared function, but just make it a macro, and be
287 // reasonably sure its name doesn't conflict with anything else
288 #define JBL_SQRD(x) ((x)*(x))
289
290 // figure out how far all the speakers are from the sound
291 curX = radius * cos(theta);
292 curY = radius * sin(theta);
293 //cout << "radius=" << radius << " curX=" << curX << " curY=" << curY << endl;
294 total_dist = 0;
295 for(i=0;i<n_channels;i++)
296 {
297 SpeakerList[i]->dist =
298 1.0 / (
299 1.0 *
300 (
301 JBL_SQRD(curX - SpeakerList[i]->x) +
302 JBL_SQRD(curY - SpeakerList[i]->y)
303 ) +
304 0.5
305 );
306 total_dist += SpeakerList[i]->dist;
307 }
308
309 // now set the interpolator entries
310 //cout << "Speaker Relative Amplitudes:" << endl;
311 for(i=0;i<n_channels;i++)
312 {
313
314 addEntryHelperFn(i, t, SpeakerList[i]->dist/total_dist);
315 //cout << "\t" << SpeakerList[i]->dist/total_dist << endl;
316 //cout << "dist=" << SpeakerList[i]->dist <<
317 //" total_dist=" << total_dist << endl;
318 }
319
320 // kill the speakers
321
322
323 for(i=0;i<n_channels;i++)
324 delete SpeakerList[i];
325 delete[] SpeakerList;
326 //the line below is the old code. it's wrong -- Ming-ching
327 //delete SpeakerList;
328}

343MultiTrack* MultiPan::spatialize_Track(Track& t, int numTracks)
344{
345 m_sample_count_type sampleCount = t.getWave().getSampleCount();
346 m_rate_type samplingRate = t.getWave().getSamplingRate();
347
348 // create an empty multi track object
349 MultiTrack *mt = new MultiTrack(numTracks, sampleCount, samplingRate);
350
351 // cycle through interpolators and tell them duration and sampleCount
352 for(int i=0; i<n_channels; i++)
353 {
354 (EnvList[i])->setSamplingRate(samplingRate);
355 (EnvList[i])->setDuration((float)sampleCount /
356 (float)samplingRate);
357 }
358
359 // get references to the input
360 SoundSample &inWave = t.getWave();
361 SoundSample &inAmp = t.getAmp();
362
363 // for each channel
364 for(int c=0; c<n_channels; c++)
365 {
366 // get an iterator for the current interpolator
367 Iterator<m_value_type> iter = (EnvList[c])->valueIterator();
368
369 // get references for this channel
370 SoundSample &thisWave = mt->get(c)->getWave();
371 SoundSample &thisAmp = mt->get(c)->getAmp();
372
373 // Iterate over each sample
374 m_value_type scale;
375 for(m_sample_count_type i=0; i<sampleCount; i++)
376 {
377 // calculate scaling factor -- FIXME, I don't
378 // understand what Pan.cpp does here, what is
379 // 'pos' for?
380 scale = iter.next();
381
382 thisWave[i] = scale * inWave[i];
383 thisAmp[i] = scale * inAmp[i];
384 }
385 }
386
387 // Return the multitrack object
388 return mt;
389}
