133 size_t note_idx = 0;
134 std::ofstream outfile;
135 outfile.open("./all_notes_flat.txt", std::ofstream::app);
136 outfile << endl << endl;
137 outfile << "ALL NOTES FLAT SECTION " <<
138 time_signature_.tempo_.getRootExactAncestor() << " : " << endl;
139 outfile << "Title: " << title << endl;
140 outfile << "Size: " << section_flat_.size() << endl;
141 for (list<Note*>::const_iterator iter = section_flat_.begin();
142 iter != section_flat_.end();
143 ++iter) {
144 const Note* note = *iter;
145 switch (note->type) {
146 case (NoteType::kBarline):
147 outfile << note_idx << " BAR ";
148 break;
149 case (NoteType::kNote):
150 outfile << note_idx << " NOTE ";
151 break;
152 case (NoteType::kRest):
153 outfile << note_idx << " REST ";
154 break;
155 case (NoteType::kTimeSignature):
156 outfile << note_idx << " TIMESIG ";
157 break;
158 case (NoteType::kUnknown):
159 outfile << note_idx << " UNKNOWN ";
160 break;
161 }
162 outfile << "start: " << note->start_t << " end: " << note->end_t << " text: " <<
163 note->type_out << " ID: " << note->rootExactAncestor << endl;
164 ++note_idx;
165 }
