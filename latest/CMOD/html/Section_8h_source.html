131 void PrintAllNotesFlat(const string& title) const {
132 size_t note_idx = 0;
133 std::ofstream outfile;
134 outfile.open("./all_notes_flat.txt", std::ofstream::app);
135 outfile << endl << endl;
136 outfile << "ALL NOTES FLAT SECTION " <<
137 time_signature_.tempo_.getRootExactAncestor() << " : " << endl;
138 outfile << "Title: " << title << endl;
139 outfile << "Size: " << section_flat_.size() << endl;
140 for (list<Note*>::const_iterator iter = section_flat_.begin();
141 iter != section_flat_.end();
142 ++iter) {
143 const Note* note = *iter;
144 switch (note->type) {
145 case (NoteType::kBarline):
146 outfile << note_idx << " BAR ";
147 break;
148 case (NoteType::kNote):
149 outfile << note_idx << " NOTE ";
150 break;
151 case (NoteType::kRest):
152 outfile << note_idx << " REST ";
153 break;
154 case (NoteType::kTimeSignature):
155 outfile << note_idx << " TIMESIG ";
156 break;
157 case (NoteType::kUnknown):
158 outfile << note_idx << " UNKNOWN ";
159 break;
160 }
161 outfile << "start: " << note->start_t << " end: " << note->end_t << " text: " <<
162 note->type_out << " ID: " << note->rootExactAncestor << endl;
163 ++note_idx;
164 }
