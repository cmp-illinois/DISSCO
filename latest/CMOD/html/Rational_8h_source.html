95 void simplifyRatio(void)
96 {
97 if(d == 0) //Indeterminate form
98 {
99 n = 0;
100 return;
101 }
102
103 if(n == 0) //Zero: assume denominator of one for consistency.
104 {
105 d = 1;
106 return;
107 }
108
109 T g = GCD(n, d);
110 while(g != 1)
111 {
112 n = n / g;
113 d = d / g;
114 g = GCD(n, d);
115 }
116 }

166 static T GCD(T a, T b)
167 {
168 //Test for zeroes.
169 if(a == 0 || b == 0)
170 return 0;
171
172 //Naturalize the numbers.
173 if(a < 0) a = -a;
174 if(b < 0) b = -b;
175
176 /*Use the iterative Euclidean algorithm. For more information, see:
177 http://en.wikipedia.org/wiki/Euclidean_algorithm#Using_iteration
178 */
179 T t;
180 while(b != 0)
181 {
182 t = b;
183 b = a % b;
184 a = t;
185 }
186 return a;
187 }
188
189 static T LCM(T a, T b)
190 {
191 //Test for zeroes.
192 if(a == 0 || b == 0)
193 return 0;
194
195 //Naturalize the numbers.
196 if(a < 0) a = -a;
197 if(b < 0) b = -b;
198
199 //Use the GCD to calculate the LCM
200 return (a * b) / GCD(a, b);
201 }
202
203 static Rational<T> Mod(Rational<T> a,
204 Rational<T> b)
205 {
206 T new_d = LCM(a.d, b.d);
207 T a_mult = new_d / a.d;
208 T b_mult = new_d / b.d;
209 a.n *= a_mult;
210 b.n *= b_mult;
211 a.d = new_d;
212 b.d = new_d;
213
214 //Now a and b have the same denominator.
215 T new_n = a.n % b.n;
216
217 return Rational<T>(new_n, new_d);
218 }

224 Rational<T> operator+(Rational<T> other)
225 {
226 if(!isDeterminate() || !other.isDeterminate())
227 return Rational<T>(0, 0);
228 T lcd = LCM(d, other.d);
229 T new_d = lcd;

573 static Rational<T> fromString(const std::string& str)
574 {
575 T len = str.length();
576 T numerator = 0;
577 T denominator = 0;
578 bool isPastSlash = false;
579 bool isNegative = false;
580
581 for(T i = 0; i < len; i++)
582 {
583 unsigned char c = str[i];
584 if(c == '-')
585 {
586 isNegative = !isNegative;
587 }
588 else if(c == '/')
589 {
590 isPastSlash = true;
591 }
592 else if(!isPastSlash)
593 {
594 if(c >= 48 && c <= 57)
595 numerator = numerator * 10 + (T)(c - 48);
596 }
597 else
598 {
599 if(c >= 48 && c <= 57)
600 denominator = denominator * 10 + (T)(c - 48);
601 }
602 }
603
604 if(!isPastSlash)
605 denominator = 1;
606 if(isNegative)
607 numerator = -numerator;
608
609 return Rational<T>(numerator, denominator);
610 }
