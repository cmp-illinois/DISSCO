159void Patter::Equivalence(int modulo, int low, int high) {
160 int newElement, numTerms, pointNum;
161 int sign = -1;
162//cout << "\t PATTER Equivalence origin:"<< this->origin << endl;
163
164 //how many more terms are possible within range/
165 //careful here, do not change: integers. Also: modulo can not be <= 0.
166 if(modulo <= 0) {
167 numTerms = 1;
168 } else {
169 numTerms = (high - origin) / modulo + 1 + (origin - low) / modulo;
170 }
171//cout << "Patter::Equivalence - modulo=" << modulo << " low=" << low <<
172// " high=" << high << " numTerms=" << numTerms << endl;
173
174 //create two more arrays now that we know their cardinal
175 vector<int> points;
176 vector<double> probs;
177
178 patty.clear(); // make sure the pattern is empty
179
180 int lastNum; // the last number to be added
181 for(unsigned location = 0; location <= intervals.size(); location++) {
182 //use current patty element as origin for next interval
183 if(location == 0) {
184 lastNum = origin;
185 } else {
186 lastNum = patty.back() + intervals[location - 1];
187 }
188
189 points.clear();
190 probs.clear();
191
192 for (int j = 0; j < numTerms; j++) { //for each possible equivalence
193
194 for (int k = 0; k < 2; k++) { //generate + - new elements
195 newElement = lastNum + modulo * j * sign;
196// cout << "newElement= " << newElement << " j=" << j <<" k=" << k << endl;
197
198 if (j != 0 || k != 1) { //only one origin !
199 if (newElement >= low && newElement <= high) {
200// cout << " newElement=" << newElement << endl;
201 points.push_back(newElement);
202 //assign probabilities according to distance from origin
203 probs.push_back( (double)(numTerms - j) / (double)numTerms );
204// cout << "prob = " << (double)(numTerms - j) / (double)numTerms << endl;
205 }
206 }
207 sign *= -1;
208 }
209 }
210
211 if (probs.size() == 0) {
212 cout << "Patter::Equivalence() error: probs array empty!" << endl;
213 }
214
215 // normalize the probability array, and order from 0 to 1
216 // (this should probably go back into utility --- cavis)
217 double partSum = 0;
218 double sum = 0;
219 for (unsigned i = 0; i < probs.size(); i++) {
220 sum += probs[i];
221 }
222
223 for (unsigned i = 0; i < probs.size(); i++) {
224 probs[i] /= sum;
225 }
226
227 for(unsigned i = 0; i < probs.size(); i++) {
228 partSum += probs[i];
229 probs[i] = partSum;
230// cout << "Patter::Equivalence - probs{" << i << "]=" << probs[i] << endl;
231 }
232
233 // choose a point at random
234 int randPosition = Random::ChooseFromProb(probs);
235 patty.push_back( points[randPosition] );
236// cout << " patty[" << location << "]=" << patty[location] << endl;
237 }
238}
