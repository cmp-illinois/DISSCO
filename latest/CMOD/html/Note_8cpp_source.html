95void Note::setPitchWellTempered(int absPitchNum) {
96 // store the pitchNum in Note
97 setPitchNum(absPitchNum);
98 pitchNum = absPitchNum;
99 octaveNum = pitchNum / 12;
100
101 octavePitch = pitchNum % 12;
102 pitchName = pitchNames[octavePitch];
103
104 pitch_out = OutNames[octavePitch];
105 string signs[8] = {",,,",",,",",","","'","''","'''","''''"};
106 string sign = signs[octaveNum];
107 pitch_out = "<" + pitch_out + sign + ">";
108
109
110 Output::addProperty("Pitch Number", pitchNum, "semitones");
111 Output::addProperty("Pitch Name", pitchName);
112 Output::addProperty("Octave Number", octaveNum);
113 Output::addProperty("Pitch In Octave", octavePitch);
114}

118int Note::HertzToPitch(float freqHz) {
119
120 int pitchNum;
121
122 if ( freqHz >= CEILING || freqHz <= MINFREQ) {
123 cerr << "WARNING: frequency out of range" << endl;
124 }
125
126 pitchNum = rint(12 * log2(freqHz / C0));
127 setPitchNum(pitchNum);
128
129 return pitchNum;
130}

166void Note::setLoudnessSones(float sones) {
167 loudnessNum = -1;
168 // cout << " sones: " << sones << endl;
169 if(sones < 0 || sones > 256) {
170 cerr << "Note received invalid value for sones!" << endl;
171 exit(1);
172 } else if(sones <= 4) {
173 loudnessMark = "ppp";
174 } else if(sones <= 8) {
175 loudnessMark = "pp";
176 } else if(sones <= 16) {
177 loudnessMark = "p";
178 } else if(sones <= 32) {
179 loudnessMark = "mp";
180 } else if(sones <= 45) {
181 loudnessMark = "mf";
182 } else if(sones <= 64) {
183 loudnessMark = "f";
184 } else if(sones <= 128) {
185 loudnessMark = "ff";
186 } else if (sones <= 256) {
187 loudnessMark = "fff";
188 }
189
190 // for particel
191 Output::addProperty("Loudness", loudnessMark);
192 loudness_out = char(92) + loudnessMark;
