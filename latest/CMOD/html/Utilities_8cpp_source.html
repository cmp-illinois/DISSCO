318string Utilities::XMLTranscode(pugi::xml_node _thisFunctionElement){
319
320 // handle empty function element and empty string
321 if (!_thisFunctionElement || !_thisFunctionElement.first_child()) {
322 return "";
323 }
324
325 // If the element has only text children, return that text directly.
326 bool hasElementChild = false;
327 for (auto c = _thisFunctionElement.first_child(); c; c = c.next_sibling()) {
328 if (c.type() == pugi::node_element) { hasElementChild = true; break; }
329 }
330 if (!hasElementChild) {
331 return std::string(_thisFunctionElement.child_value());
332 }
333
334 // Otherwise serialize each child node (XML for elements, raw text for text nodes).
335 std::ostringstream oss;
336 for (auto c = _thisFunctionElement.first_child(); c; c = c.next_sibling()) {
337 c.print(oss, "", pugi::format_raw | pugi::format_no_declaration);
338 }
339 return oss.str();
340}

352double Utilities::evaluate(std::string _input, void* _object){
353 if (_input == "") return 0;
354
355 string workingString = _input;
356 // Test if there is any function in this string (look for <Fun>), if so,
357 // replace the function with the evaluated number. Repeat until all the
358 // functions are replaced by numbers.
359 size_t locOfFun = workingString.find("<Fun>");
360 int functionStringLength;
361 while (locOfFun != string::npos){
362 size_t locOfEndFun = findTheEndOfFirstFunction (workingString);
363 functionStringLength = ((int) locOfEndFun) + 6 - ((int) locOfFun);//6 is the length of "</fun>"
364 string functionString = workingString.substr(locOfFun, functionStringLength);
365 string evaluatedFunction = evaluateFunction( functionString, _object);
366 string front = workingString.substr(0, locOfFun);
367 string back = workingString.substr(((int)locOfEndFun) +6);
368 workingString = front + evaluatedFunction + back;
369// cout << "utilities string: " << workingString << endl;
370 // look for next function
371 locOfFun = workingString.find("<Fun>");
372 }
373 // evaluate the expression to the final result
374 mu::Parser p;
375 p.SetExpr(workingString);
376 double result;
377
378 try {
379 result = p.Eval();
380//cout << "utilities result: " << result << endl;
381 } catch (mu::ParserError) {
382 cerr << "Oooops, we find a typo in your project." << endl;
383 if (_object != NULL){
384 cerr << "The string we see is located in Event (or Bottom): " << ((Event*)_object)->getEventName() << endl;
385 }
386 cerr << "The string we see is: " << _input << endl;
387 exit(1);
388 }
389
390 return result;
391}

447std::vector<std::string> Utilities::listElementToStringVector(
448 pugi::xml_node _listElement){
449 std::vector<std::string> list;
450 string listString = XMLTranscode(_listElement);
451 bool doneProcessingListString = false;
452 while (!doneProcessingListString){
453 // Step 1: Isolate the first element of the list and put its string
454 // representation in the vector.
455 bool listElementLocated = false;
456 size_t locationOfComma = listString.find(","); //this can be npos
457 size_t locationOfFun = listString.find("<Fun>"); //this can be npos
458 while (!listElementLocated){
459
460 if (locationOfComma == string::npos){ //reach the end of the string
461 listElementLocated =true;
462 }
463 else if (locationOfFun==string::npos || locationOfFun> locationOfComma){
464 listElementLocated = true;
465 }
466 else { //find the end of this fun, and find comma and fun again
467 size_t endOfThisFun =Utilities::findTheEndOfFirstFunction(listString.substr(locationOfFun)) + locationOfFun;
468
469 locationOfComma = listString.find(",", endOfThisFun);
470 locationOfFun = listString.find("<Fun>",endOfThisFun);
471 }
472
473 } // end of inner while
474
475 // Step 2: Cut the string from the listString
476 if (locationOfComma ==string::npos){
477 list.push_back(listString);
478 }
479 else {
480 string thisListElement = listString.substr(0, locationOfComma);
481 list.push_back(thisListElement);
482 listString = listString.substr(locationOfComma +1);
483 }
484 // Step 3: check if the listString is empty
485 if (locationOfComma ==string::npos){
486 doneProcessingListString = true;
487 }
488 }
489
490 return list;
491}

495//this function assume that at least 1 "<Fun>" exists;
496size_t Utilities::findTheEndOfFirstFunction(string _input){
497
498 int depth = 1; // The depth of nested functions
499
500 // When seeing a <Fun>, depth ++, seeing a </Fun>, depth --
501 // When depth = 0, the outermost function has ended.
502 size_t location = _input.find("<Fun>");
503 while (depth!=0){
504 size_t nextFun = _input.find("<Fun>", location + 1);
505 size_t nextEndFun = _input.find("</Fun>", location + 1);
506
507 if (nextFun < nextEndFun){
508 depth ++;
509 location = nextFun;
510 }
511 else {
512 depth --;
513 location = nextEndFun;
514 }
515 }
516 return location;
517}
