79void Tempo::setTempo(string newTempo) {
80 int length = (int)newTempo.length();
81 string firstHalf, secondHalf;
82 bool firstHalfIsNumber = false;
83 bool inSecondHalf = false;
84 bool countDigits = false;
85 bool encounteredSecondHalfSlash = false;
86 int digitCount = 0;
87
88 for(int i = 0; i < length; i++) {
89 char c = newTempo[i];
90 if(c == ' ')
91 continue;
92 else if(c == '=')
93 inSecondHalf = true;
94 else if(!inSecondHalf) {
95 if((c >= '0' && c <= '9') || c == '/')
96 firstHalfIsNumber = true;
97 firstHalf.append(1, c);
98 }
99 else {
100 if(c == '.')
101 countDigits = true;
102 else if(c >= '0' && c <= '9') {
103 secondHalf.append(1, c);
104 if(countDigits)
105 digitCount++;
106 }
107 else if(c == '/' && !countDigits && !encounteredSecondHalfSlash) {
108 encounteredSecondHalfSlash = true;
109 secondHalf.append("/");
110 }
111 }
112 }
113
114 //A really obscure way to add /10 /100 /1000 etc. depending on digitCount
115 if(countDigits) {
116 secondHalf.append("/1");
117 secondHalf.append(digitCount, '0');
118 }
119 tempoBeatsPerMinute = secondHalf;
120
121 if(firstHalfIsNumber) {
122 tempoBeat = firstHalf;
123 return;
124 }
125
126 if(firstHalf.find("thirt") != string::npos)
127 tempoBeat = Ratio(1,32);
128 else if(firstHalf.find("six") != string::npos)
129 tempoBeat = Ratio(1,16);
130 else if(firstHalf.find("eig") != string::npos)
131 tempoBeat = Ratio(1,8);
132 else if(firstHalf.find("quar") != string::npos)
133{ tempoBeat = Ratio(1,4);
134}
135 else if(firstHalf.find("hal") != string::npos)
136 tempoBeat = Ratio(1,2);
137 else if(firstHalf.find("who") != string::npos)
138 tempoBeat = Ratio(1,1);
139
140 if(firstHalf.find("doub") != string::npos)
141 tempoBeat *= Ratio(7,4);
142 else if(firstHalf.find("dot") != string::npos)
143 tempoBeat *= Ratio(3,2);
144
145 if(firstHalf.find("tripl") != string::npos)
146 tempoBeat *= Ratio(2,3);
147 else if(firstHalf.find("quin") != string::npos)
148 tempoBeat *= Ratio(4,5);
149 else if(firstHalf.find("sext") != string::npos)
150 tempoBeat *= Ratio(5,6);
151 else if(firstHalf.find("sept") != string::npos)
152 tempoBeat *= Ratio(6,7);
153}

157void Tempo::setTimeSignature(string newTimeSignature) {
158 int length = (int)newTimeSignature.length();
159 string firstHalf, secondHalf;
160 bool inSecondHalf = false;
161 for(int i = 0; i < length; i++) {
162 char c = newTimeSignature[i];
163 if(c == ' ')
164 continue;
165 else if(c == '/')
166 inSecondHalf = true;
167 else if(c >= '0' && c <= '9') {
168 if(!inSecondHalf) {
169 firstHalf.append(1, c);
170 }
171 else {
172 secondHalf.append(1, c);
173 }
174 }
175 }
176 timeSignatureBeatsPerBar = Ratio(firstHalf);
177 timeSignatureBeat = Ratio(1) / Ratio(secondHalf);
178}

308int Tempo::convertSecondsToEDUs(float seconds) {
309 float beats_per_second = tempoBeatsPerMinute.To<float>() / 60.0f;
310 float beats = seconds * beats_per_second;
311 return static_cast<int>(lround(beats * EDUPerTimeSignatureBeat.To<float>()));
