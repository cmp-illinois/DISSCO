54Modifier::Modifier(const Modifier& orig) {
55 type = orig.type;
56 applyHow = orig.applyHow;
57 checkPt = orig.checkPt;
58 partialNum = orig.partialNum;
59 partialResultString = orig.partialResultString;
60
61 if (orig.probEnv != NULL) {
62 probEnv = new Envelope(*orig.probEnv);
63 } else {
64 probEnv = NULL;
65 }
66
67 for (unsigned i = 0; i < orig.env_values.size(); i++) {
68 if (orig.env_values[i] != NULL) {
69 env_values.push_back(new Envelope(*orig.env_values[i]));
70 } else {
71 env_values.push_back(NULL);
72 }
73 }
74}

78Modifier& Modifier::operator=(const Modifier& rhs) {
79 if (this == &rhs) {
80 return *this;
81 }
82 type = rhs.type;
83 applyHow = rhs.applyHow;
84 checkPt = rhs.checkPt;
85 partialNum = rhs.partialNum;
86 partialResultString = rhs.partialResultString;
87
88 delete probEnv;
89 if (rhs.probEnv != NULL) {
90 probEnv = new Envelope(*rhs.probEnv);
91 } else {
92 probEnv = NULL;
93 }
94 for (auto* env : env_values) {
95 delete env;
96 }
97 env_values.clear();
98 for (unsigned i = 0; i < rhs.env_values.size(); i++) {
99 if (rhs.env_values[i] != NULL) {
100 env_values.push_back(new Envelope(*rhs.env_values[i]));
101 } else {
102 env_values.push_back(NULL);
103 }
104 }
105 return *this;

146float Modifier::getProbability(double checkPoint) {

148 cerr << "Modifier::getProbability -- Error: checkPt out of bounds;" << endl;

176bool Modifier::willOccur(double checkPoint) {
177 bool result = false;
178 double rand = Random::Rand();
179
180 if (rand > 0.0 && rand <= getProbability(checkPoint)) {
181 result = true;
182 }
183 return result;
184}

198void Modifier::applyModSound(Sound* snd) {
199 if (type == "FREQUENCY" || type == "GLISSANDO"
200 || type == "BEND") {
201 snd->setPartialParam(FREQ_ENV, *(env_values[0]));
202 } else if (type == "TREMOLO") {
203 snd->setPartialParam(TREMOLO_AMP, *(env_values[0]));
204 snd->setPartialParam(TREMOLO_RATE, *(env_values[1]));
205 } else if (type == "VIBRATO") {
206 snd->setPartialParam(VIBRATO_AMP, *(env_values[0]));
207 snd->setPartialParam(VIBRATO_RATE, *(env_values[1]));
208 } else if (type == "PHASE") {
209 snd->setPartialParam(PHASE, *(env_values[0]));
210 } else if (type == "AMPTRANS") {
211 snd->setPartialParam(AMPTRANS_AMP_ENV, *(env_values[0]));
212 snd->setPartialParam(AMPTRANS_RATE_ENV, *(env_values[1]));
213 snd->setPartialParam(AMPTRANS_WIDTH, *(env_values[2]));
214 } else if (type == "FREQTRANS") {
215 snd->setPartialParam(FREQTRANS_AMP_ENV, *(env_values[0]));
216 snd->setPartialParam(FREQTRANS_RATE_ENV, *(env_values[1]));
217 snd->setPartialParam(FREQTRANS_WIDTH, *(env_values[2]));
218 } else if (type == "WAVE_TYPE") {
219 snd->setPartialParam(WAVE_TYPE, env_values[0]->getValue(checkPt, 1));
220 } else if (type == "DETUNE"){
221 snd->setDetune(direction, spread, velocity);
222 }else {
223 cerr << "ERROR: Modifier given an invalid type: " << type << endl;
224 exit(1);
225 }
226}
