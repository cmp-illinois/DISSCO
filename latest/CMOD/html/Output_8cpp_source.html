76// find "name" in all properties and return its value
77string OutputNode::getProperty(string name) {
78 for(unsigned int i = 0; i < propertyNames.size(); i++)
79 if(sanitize(propertyNames[i]) == sanitize(name))
80 return propertyValues[i];
81 return "";

87string OutputNode::getXML(void) {
88 string s;
89 s += "<";
90 s += sanitize(nodeName);
91 s += ">\n";
92
93 for(unsigned i = 0; i < propertyNames.size(); i++) {
94 s += "<";
95 s += sanitize(propertyNames[i]);
96 s += ">";
97
98 s += propertyValues[i];
99
100 s += "</";
101 s += sanitize(propertyNames[i]);
102 s += ">\n";
103 }
104
105 for(unsigned i = 0; i < subNodes.size(); i++) {
106 if(subNodes[i]->isBuildPhase())
107 continue;
108 s += subNodes[i]->getXML();
109 }
110
111 s += "</";
112 s += sanitize(nodeName);
113 s += ">\n";
114 return s;
115}

145string Output::getLevelIndentation(bool isProperty, bool isEndLevel) {
146 string indentation;

201void Output::beginSubLevel(string name) {
202 if(!top)
203 top = new OutputNode(name);
204 else
205 getCurrentLevelNode()->subNodes.push_back(new OutputNode(name));
206 level++;
207
208 //Immediately write level to particel.
209 writeLineToParticel(getLevelIndentation(false, false) +
210 getCurrentLevelNode()->nodeName);
211}

215
216void Output::addProperty(string name, string value, string units) {
217 OutputNode* current = getCurrentLevelNode();
218 if(!current)
219 cerr << "Warning: Top level does not exist. Property can not be added."
220 " name=" << name << endl;
221 else
222 current->addProperty(name, value, units);
223
224 //Immediately write property to particel.
225 string stringToWrite = getPropertyIndentation() + name + ": " + value;
226 if(units != "")
227 stringToWrite = stringToWrite + " " + units;
228 writeLineToParticel(stringToWrite);
229}

237 getCurrentLevelNode()->nodeName);
