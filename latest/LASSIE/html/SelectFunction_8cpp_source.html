102// Split on commas that aren't inside a <Fun>...</Fun> block. Matches the
103// legacy parser's behavior so nested function strings round-trip cleanly.
104QStringList SelectFunction::splitTopLevelCommas(const QString& s) {
105 QStringList out;
106 int start = 0;
107 int depth = 0;
108 int i = 0;
109 while (i < s.size()) {
110 if (s.mid(i, 5) == QStringLiteral("<Fun>")) { ++depth; i += 5; continue; }
111 if (s.mid(i, 6) == QStringLiteral("</Fun>")) { --depth; i += 6; continue; }
112 if (depth == 0 && s[i] == QChar(',')) {
113 out << s.mid(start, i - start).trimmed();
114 start = i + 1;
115 }
116 ++i;
117 }
118 if (start < s.size()) out << s.mid(start).trimmed();
119 return out;
120}
