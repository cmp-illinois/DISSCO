75Modifier& Modifiers::getBackendLayer() {
76 ProjectManager* pm = Inst::get_project_manager();
77
78 if (m_eventType != bottom) {
79 HEvent* hevent = nullptr;
80 if (m_eventType == top)
81 hevent = &pm->topevent();
82 else if (m_eventType == high)
83 hevent = &pm->highevents()[m_eventIndex];
84 else if (m_eventType == mid)
85 hevent = &pm->midevents()[m_eventIndex];
86 else if (m_eventType == low)
87 hevent = &pm->lowevents()[m_eventIndex];
88
89 return hevent->modifiers[m_modifierIndex];
90 } else {
91 ExtraInfo* bevent = &pm->bottomevents()[m_eventIndex].extra_info;
92 return bevent->modifiers[m_modifierIndex];
93 }

186void Modifiers::updateModState() {

195 : tr("Rate Value Envelope:"));

197 // GLISSANDO (2) is always SOUND — disable the apply combo

199 /*if (typeIndex == 2) {
200 ui->modifierApply->blockSignals(true);
201 ui->modifierApply->setCurrentIndex(0);
202 ui->modifierApply->blockSignals(false);
203 ui->modifierApply->setEnabled(false);
204 } else {
205 ui->modifierApply->setEnabled(true);

207
208 bool isPartial = (ui->modifierApply->currentIndex() == 1);
209
210 // Enabled flags per type: { prob, mag, rate, width, spread, dir, vel, res }
211 // res (partial result string) depends on apply mode, not type — set below

214 const Row rows[8] = {

217 { ui->modifierRateLabel, ui->modifierRateEdit, ui->modifierRateFunButton },

221 { ui->modifierVelLabel, ui->modifierVelEdit, ui->modifierVelFunButton },
222 { ui->modifierResLabel, ui->modifierResEdit, ui->modifierResFunButton },
223 };
224
225 if (typeIndex < 0 || typeIndex >= modifierTypeCount) {
226 ui->modifierApply->setEnabled(false);
227 for (const Row& row : rows) {

238 rows[i].label->setEnabled(enabled);

281void Modifiers::modFunctionButtonClicked(ModChanged type) {
282 QLineEdit* target = nullptr;
283 FunctionGenerator* gen = nullptr;
284
285 switch (type) {
286 case modProbabilityChanged:
287 target = ui->modifierProbEdit;
288 break;
289 case modMagnitudeChanged:
290 target = ui->modifierMagEdit;
291 break;
292 case modRateChanged:
293 target = ui->modifierRateEdit;
294 break;
295 case modWidthChanged:
296 target = ui->modifierWidthEdit;
297 break;
298 case modPartialChanged:
299 target = ui->modifierResEdit;
300 break;
301 case modSpreadChanged:
302 target = ui->modifierSpreadEdit;
303 break;
304 case modDirChanged:
305 target = ui->modifierDirEdit;
306 break;
307 case modVelChanged:
308 target = ui->modifierVelEdit;
309 break;
310 case modNameChanged:
311 break;
312 default:
313 break;
314 }
315
316 if (!target) return;
317
318 // The structured editor is deliberately PHASE_MOD-only. Other modifier
319 // types retain their legacy FunctionGenerator path and wire semantics.
320 if (type == modPartialChanged
321 && currentModifierType(ui->modifierType) == phaseModType) {
322 ProjectManager* pm = Inst::get_project_manager();
323 int spectrumPartialCount = 0;
324 constexpr int generatedSpectrumPartialCount = 20;
325 if (pm->get_curr_project()) {
326 for (const SpectrumEvent& spectrum : pm->spectrumevents()) {
327 spectrumPartialCount = std::max(
328 spectrumPartialCount,
329 static_cast<int>(spectrum.spectrum.partials.size()));
330
331 bool isInteger = false;
332 const int declaredCount = spectrum.num_partials.toInt(&isInteger);
333 if (isInteger)
334 spectrumPartialCount = std::max(spectrumPartialCount, declaredCount);
335
336 // CMOD's generated-spectrum path currently creates 20 partials.
337 if (!spectrum.generate_spectrum.trimmed().isEmpty())
338 spectrumPartialCount = std::max(spectrumPartialCount,
339 generatedSpectrumPartialCount);
340 }
341 }
342
343 PartialModifierDialog dialog(this,
344 std::max(1, spectrumPartialCount),
345 target->text());
346 if (dialog.exec() == QDialog::Accepted)
347 target->setText(dialog.resultString());
348 return;
349 }
350
351 gen = new FunctionGenerator(nullptr, functionReturnENV, target->text());
352 if (gen) {
353 if (gen->exec() == QDialog::Accepted && !gen->getResultString().isEmpty())
354 target->setText(gen->getResultString());
355 delete gen;
356 }
357}
