191QStandardItem* LayerBox::extractItemFromDrop(QDropEvent* event)
192{
193 const QMimeData* mime = event->mimeData();
194 if (!mime->hasFormat("application/x-qabstractitemmodeldatalist")) {
195 qDebug() << "extractItemFromDrop: MIME format not matched";
196 return nullptr;
197 }
198
199 QByteArray encoded = mime->data("application/x-qabstractitemmodeldatalist");
200 QDataStream stream(&encoded, QIODevice::ReadOnly);
201
202 QString foundType;
203 QString foundName;
204 bool gotType = false;
205 bool gotName = false;
206
207 while (!stream.atEnd()) {
208 int row, col;
209 QMap<int, QVariant> roleDataMap;
210 stream >> row >> col >> roleDataMap;
211
212 QString roleType = roleDataMap.value(Qt::UserRole + 1).toString();
213 QString roleName = roleDataMap.value(Qt::UserRole + 2).toString();
214 if (!roleType.isEmpty()) { foundType = roleType; gotType = true; }
215 if (!roleName.isEmpty()) { foundName = roleName; gotName = true; }
216 if (gotType && gotName) break;
217 }
218
219 if (!gotType && !gotName) {
220 qDebug() << "extractItemFromDrop: no usable data found";
221 return nullptr;
222 }
223
224 QStandardItem* item = new QStandardItem;
225 item->setText(gotName ? foundName : foundType);
226 item->setData(foundType, Qt::UserRole + 1);
227 item->setData(foundName, Qt::UserRole + 2);
228 return item;
229}
