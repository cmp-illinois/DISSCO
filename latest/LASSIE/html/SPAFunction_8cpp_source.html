174void SPAFunction::populateFromXML(QXmlStreamReader& reader) {
175 const QString method = nextChildInner(reader);
176 const QString apply = nextChildInner(reader);
177
178 // Block radio signals so setChecked doesn't trigger another rebuild
179 // for each toggle; we'll rebuild once explicitly below.
180 {
181 QSignalBlocker b1(m_stereo);
182 QSignalBlocker b2(m_multiPan);
183 QSignalBlocker b3(m_polar);
184 QSignalBlocker b4(m_applySound);
185 QSignalBlocker b5(m_applyPartial);
186 if (method == QStringLiteral("MULTI_PAN")) m_multiPan->setChecked(true);
187 else if (method == QStringLiteral("POLAR")) m_polar->setChecked(true);
188 else m_stereo->setChecked(true);
189 if (apply == QStringLiteral("PARTIAL")) m_applyPartial->setChecked(true);
190 else m_applySound->setChecked(true);
191 }
192 rebuildChannelsForCurrentMethod();
193
194 if (reader.readNextStartElement()) { // <Channels>
195 int chanIdx = 0;
196 while (reader.readNextStartElement()) { // <Partials>
197 while (chanIdx >= m_channels.size())
198 insertChannel(m_channels.isEmpty() ? nullptr : m_channels.last());
199 auto* chan = m_channels[chanIdx];
200 int rowIdx = 0;
201 while (reader.readNextStartElement()) { // <P>
202 while (rowIdx >= chan->rowCount())
203 chan->addRow(chan->rowCount(), true);
204 chan->setRowText(rowIdx, readInner(reader));
205 ++rowIdx;
206 }
207 ++chanIdx;
208 }
209 }
210}
