28 inline QString readInner(QXmlStreamReader& r) {
29 QString result;
30 QXmlStreamWriter w(&result);
31 bool hasChildElement = false;
32 QString textBuffer;
33 int depth = 0;
34 while (!r.atEnd()) {
35 r.readNext();
36 switch (r.tokenType()) {
37 case QXmlStreamReader::StartElement: {
38 hasChildElement = true;
39 w.writeStartElement(r.name().toString());
40 const auto attrs = r.attributes();
41 for (const auto& a : attrs)
42 w.writeAttribute(a.name().toString(), a.value().toString());
43 ++depth;
44 break;
45 }
46 case QXmlStreamReader::EndElement:
47 if (depth == 0)
48 return hasChildElement ? result : textBuffer;
49 w.writeEndElement();
50 --depth;
51 break;
52 case QXmlStreamReader::Characters:
53 if (hasChildElement)
54 w.writeCharacters(r.text().toString());
55 else
56 textBuffer += r.text().toString();
57 break;
58 default:
59 break;
60 }
61 }
62 return hasChildElement ? result : textBuffer;
63 }

269 // r at <Event> StartElement (with orderInPalette attribute).
270 QString orderinpalette = r.attributes().value("orderInPalette").toString();
271
272 // First child is <EventType>, whose text content is the integer type.
273 if (!r.readNextStartElement()) { r.skipCurrentElement(); return; }
274 int typeInt = QtParser::readInner(r).toInt();
275 Eventtype type = (Eventtype)typeInt;
276
277 switch (type) {
278 case top:
279 case high:
280 case mid:
281 case low: {
282 HEvent eh;
283 eh.orderinpalette = orderinpalette;
284 eh.type = type;
285 QtParser::parseHEventChildren(r, eh);
286 qDebug() << "parsed" << eh.type << "event named" << eh.name;
287 switch (type) {
288 case top: top_event = eh; break;
289 case high: high_events.append(eh); break;
290 case mid: mid_events.append(eh); break;
291 case low: low_events.append(eh); break;
292 default: break;
293 }
294 break;
295 }
296 case bottom: {
297 BottomEvent eb;
298 eb.event.orderinpalette = orderinpalette;
299 eb.event.type = type;
300 QtParser::parseBottomEventChildren(r, eb);
301 qDebug() << "parsed Bottom event named" << eb.event.name;
302 bottom_events.append(eb);
303 break;
304 }
305 case sound: {
306 SpectrumEvent espec;
307 espec.orderinpalette = orderinpalette;
308 QtParser::parseSpectrumEventChildren(r, espec);
309 qDebug() << "parsed Spectrum event named" << espec.name;
310 spectrum_events.append(espec);
311 break;
312 }
313 case note: {
314 NoteEvent en;
315 en.orderinpalette = orderinpalette;
316 QtParser::parseNoteEventChildren(r, en);
317 qDebug() << "parsed Note event named " << en.name;
318 note_events.append(en);
319 break;
320 }
321 case env: {
322 EnvelopeEvent ee;
323 ee.orderinpalette = orderinpalette;
324 ee.name = QtParser::nextChildInner(r);
325 ee.envelope_builder = QtParser::nextChildInner(r);
326 QtParser::consumeRest(r);
327 qDebug() << "parsed Envelope event named" << ee.name;
328 envelope_events.append(ee);
329 break;
330 }
331 case sieve: {
332 SieveEvent esi;
333 esi.orderinpalette = orderinpalette;
334 esi.name = QtParser::nextChildInner(r);
335 esi.sieve_builder = QtParser::nextChildInner(r);

338 sieve_events.append(esi);

340 }
341 case spa: {
342 SpaEvent espa;
343 espa.orderinpalette = orderinpalette;
344 espa.name = QtParser::nextChildInner(r);
345 espa.spatialization = QtParser::nextChildInner(r);
346 QtParser::consumeRest(r);
347 qDebug() << "parsed Spa event named" << espa.name;
348 spa_events.append(espa);

350 }
351 case pattern: {
352 PatternEvent ep;
353 ep.orderinpalette = orderinpalette;
354 ep.name = QtParser::nextChildInner(r);
355 ep.pattern_builder = QtParser::nextChildInner(r);
356 QtParser::consumeRest(r);
357 qDebug() << "parsed Pattern event named" << ep.name;
358 pattern_events.append(ep);
359 break;
360 }
361 case reverb: {
362 ReverbEvent er;
363 er.orderinpalette = orderinpalette;
364 er.name = QtParser::nextChildInner(r);

387void ProjectManager::parse(Project* p, const QString& filepath) {
388 QFile file(filepath);
389 if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
390 qDebug() << "ERROR: cannot open" << filepath;
391 return;
392 }
393 QXmlStreamReader r(&file);
394
395 if (!r.readNextStartElement()) return; // <DISSCO> root
396
397 while (r.readNextStartElement()) {
398 const QString elName = r.name().toString();
399
400 if (elName == "ProjectConfiguration") {
401 // Children: <Title>, <FileFlag>, <TopEvent>, <PieceStartTime>, <Duration>,
402 // <Synthesis>, <Score>, <Grandstaff>, <NumOfStaff>, <NumOfChannels>,
403 // <SampleRate>, <SampleSize>, <NumOfThreads>, <OutputParticel> (optional)
404 r.readNextStartElement(); // <Title> — skipped
405 QtParser::readInner(r);
406 p->file_flag = QtParser::nextChildInner(r);
407 QtParser::nextChildInner(r); // <TopEvent> — always "0"
408 p->top_event.name = "0";
409 p->start_time = QtParser::nextChildInner(r);
410 p->duration = QtParser::nextChildInner(r);
411 p->synthesis = (QtParser::nextChildInner(r) == "True");
412 p->score = (QtParser::nextChildInner(r) == "True");
413 p->grand_staff = (QtParser::nextChildInner(r) == "True");
414 p->num_staffs = QtParser::nextChildInner(r);
415 p->num_channels = QtParser::nextChildInner(r);
416 p->sample_rate = QtParser::nextChildInner(r);
417 p->sample_size = QtParser::nextChildInner(r);
418 p->num_threads = QtParser::nextChildInner(r);
419 // <OutputParticel> is optional; defaults to false.
420 p->output_particel = false;
421 while (r.readNextStartElement()) {
422 if (r.name() == QStringView(u"OutputParticel"))
423 p->output_particel = (QtParser::readInner(r) == "True");
424 else
425 r.skipCurrentElement();
426 }
427 }
428 else if (elName == "NoteModifiers") {
429 // First child: default modifiers bitstring (currently ignored — see
430 // pre-Qt code: the loop that consumed it was commented out).
431 if (r.readNextStartElement()) {
432 QtParser::readInner(r);
433 // Second child: custom modifiers list
434 if (r.readNextStartElement()) {
435 while (r.readNextStartElement())
436 p->custom_note_modifiers.append(QtParser::readInner(r));
437 }
438 QtParser::consumeRest(r);
439 }
440 qDebug() << "Passed modifiers";
441 }
442 else if (elName == "EnvelopeLibrary") {
443 QString envLibText = r.readElementText(QXmlStreamReader::IncludeChildElements);
444 EnvelopeLibrary* envelopeLibrary = new EnvelopeLibrary();
445 if (!envLibText.isEmpty()) {
446 QString tempPath = filepath + ".lib.temp";
447 QFile temp(tempPath);
448 if (temp.open(QIODevice::WriteOnly | QIODevice::Text)) {
449 QTextStream out(&temp);
450 out << envLibText;
451 temp.close();
452 envelopeLibrary->loadLibraryNewFormat(
453 const_cast<char*>(tempPath.toLocal8Bit().constData()));
454 QFile::remove(tempPath);
455 }
456 }
457 EnvelopeLibraryEntry* previousEntry = nullptr;
458 for (int i = 1; i <= envelopeLibrary->size(); i++) {
459 Envelope* thisEnvelope = envelopeLibrary->getEnvelope(i);
460 EnvelopeLibraryEntry* thisEntry = new EnvelopeLibraryEntry(thisEnvelope, i);
461 delete thisEnvelope;
462 if (previousEntry == nullptr) {
463 p->elentry = thisEntry;
464 thisEntry->prev = nullptr;
465 } else {
466 previousEntry->next = thisEntry;
467 thisEntry->prev = previousEntry;
468 }
469 previousEntry = thisEntry;
470 }
471 delete envelopeLibrary;
472 qDebug() << "Passed envelopes";
473 }
474 else if (elName == "MarkovModelLibrary") {
475 QString data = r.readElementText(QXmlStreamReader::IncludeChildElements);
476 if (!data.isEmpty()) {
477 std::stringstream ss(data.toStdString());
478 long long size = 0;
479 ss >> size;
480 p->markov_models.resize(size);
481 std::string line;
482 std::getline(ss, line, '\n');
483 for (long long i = 0; i < size; i++) {
484 std::string modelText;
485 std::getline(ss, line, '\n'); modelText = line + '\n';
486 std::getline(ss, line, '\n'); modelText += line + '\n';
487 std::getline(ss, line, '\n'); modelText += line + '\n';
488 std::getline(ss, line, '\n'); modelText += line;
489 p->markov_models[i].from_str(modelText);
490 }
491 }
492 qDebug() << "Passed markov";
493 }
494 else if (elName == "Events") {
495 while (r.readNextStartElement()) {
496 if (r.name() == QStringView(u"Event"))
497 p->parseEvent(r);
498 else
499 r.skipCurrentElement();
500 }
501 }
502 else {
503 r.skipCurrentElement();
504 }
505 }
506
507 if (r.hasError())
508 qDebug() << "XML parse error:" << r.errorString();
509}
