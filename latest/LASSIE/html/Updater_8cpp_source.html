36// Strip an optional leading 'v' and parse. QVersionNumber::fromString
37// stops at the first non-numeric character, so pre-release/build
38// suffixes ("-rc1", "-test", "+abc") fall away naturally.
39QVersionNumber parseTag(const QString &tag) {
40 QString trimmed = tag.trimmed();
41 if (trimmed.startsWith(QLatin1Char('v')) || trimmed.startsWith(QLatin1Char('V'))) {
42 trimmed.remove(0, 1);
43 }
44 return QVersionNumber::fromString(trimmed);
45}

47// Asset filename substring that identifies this platform's installer.
48// Must stay in sync with packaging/Packaging.cmake's CPACK_PACKAGE_FILE_NAME

111void Updater::onReleaseReply(QNetworkReply *reply) {
112 reply->deleteLater();
113
114 if (reply->error() != QNetworkReply::NoError) {
115 // 404 from /releases/latest just means no published release exists —
116 // common during early-stage projects. Treat as "up to date".
117 const int status = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
118 if (status == 404) {
119 if (trigger_ == Trigger::Manual) {
120 QMessageBox::information(uiParent_, tr("Check for Updates"),
121 tr("You're up to date.\n\nInstalled: %1\n\n"
122 "No published releases on GitHub yet.")
123 .arg(currentVersion().toString()));
124 }
125 return;
126 }
127 if (trigger_ == Trigger::Manual) {
128 reportError(tr("Couldn't reach GitHub: %1").arg(reply->errorString()));
129 }
130 return;
131 }
132
133 const QJsonObject obj = QJsonDocument::fromJson(reply->readAll()).object();
134
135 ReleaseInfo info;
136 info.tagName = obj.value("tag_name").toString();
137 info.version = parseTag(info.tagName);
138 info.releaseNotes = obj.value("body").toString();
139 info.htmlUrl = QUrl(obj.value("html_url").toString());
140
141 if (info.version.isNull()) {
142 if (trigger_ == Trigger::Manual) {
143 reportError(tr("Couldn't parse latest release tag '%1'.").arg(info.tagName));
144 }
145 return;
146 }
147
148 if (info.version <= currentVersion()) {
149 if (trigger_ == Trigger::Manual) {
150 QMessageBox::information(uiParent_, tr("Check for Updates"),
151 tr("You're up to date.\n\nInstalled: %1\nLatest released: %2")
152 .arg(currentVersion().toString(), info.version.toString()));
153 }
154 return;
155 }
156
157 // Honour a prior "skip this version" choice on background checks.
158 if (trigger_ == Trigger::Auto &&
159 QSettings().value(kSkipVersionKey).toString() == info.version.toString()) {
160 return;
161 }
162
163 // Find the asset matching this OS.
164 const QString marker = platformAssetMarker();
165 if (!marker.isEmpty()) {
166 const QString archHint = QSysInfo::currentCpuArchitecture();
167 const QJsonArray assets = obj.value("assets").toArray();
168 for (const QJsonValue &v : assets) {
169 const QJsonObject a = v.toObject();
170 const QString name = a.value("name").toString();
171 if (!name.contains(marker)) continue;
172#if defined(Q_OS_LINUX)
173 // Prefer an arch-matched build when multiple AppImages exist.
174 if (!name.contains(archHint) && assets.size() > 1) continue;
175#endif
176 info.assetName = name;
177 info.assetUrl = QUrl(a.value("browser_download_url").toString());
178 info.assetSize = static_cast<qint64>(a.value("size").toDouble());
179 break;
180 }
181 }
182
183 if (!info.assetUrl.isValid()) {
184 QMessageBox::information(uiParent_, tr("Check for Updates"),
185 tr("Version %1 is available, but no installer was published for "
186 "this platform. Opening the release page.")
187 .arg(info.version.toString()));
188 QDesktopServices::openUrl(info.htmlUrl);
189 return;
190 }
191
192 presentReleaseDialog(info);
193}

224void Updater::downloadAndInstall(const ReleaseInfo &info) {
225 const QString updatesDir = QStandardPaths::writableLocation(QStandardPaths::CacheLocation)
226 + QStringLiteral("/updates");
227 QDir().mkpath(updatesDir);
228 const QString localPath = updatesDir + QLatin1Char('/') + info.assetName;
229
230 // Reuse a prior download if it's the right size.
231 if (info.assetSize > 0 && QFileInfo(localPath).size() == info.assetSize) {
232 launchInstaller(localPath);
233 return;
234 }
235
236 QFile *file = new QFile(localPath);
237 if (!file->open(QIODevice::WriteOnly | QIODevice::Truncate)) {
238 reportError(tr("Couldn't write to %1: %2").arg(localPath, file->errorString()));
239 delete file;
240 return;
241 }
242
243 QNetworkRequest req(info.assetUrl);
244 req.setRawHeader("User-Agent", userAgent());
245 req.setAttribute(QNetworkRequest::RedirectPolicyAttribute,
246 QNetworkRequest::NoLessSafeRedirectPolicy);
247 QNetworkReply *reply = net_->get(req);
248
249 QProgressDialog *progress = new QProgressDialog(
250 tr("Downloading %1…").arg(info.assetName),
251 tr("Cancel"), 0, 100, uiParent_);
252 progress->setWindowModality(Qt::WindowModal);
253 progress->setMinimumDuration(0);
254 progress->setAutoClose(true);
255
256 connect(reply, &QNetworkReply::downloadProgress, progress,
257 [progress](qint64 received, qint64 total) {
258 if (total > 0) {
259 progress->setMaximum(100);
260 progress->setValue(static_cast<int>((received * 100) / total));
261 } else {
262 progress->setMaximum(0); // indeterminate
263 }
264 });
265 connect(reply, &QNetworkReply::readyRead, file, [reply, file] {
266 file->write(reply->readAll());
267 });
268 connect(progress, &QProgressDialog::canceled, reply, &QNetworkReply::abort);
269
270 connect(reply, &QNetworkReply::finished, this,
271 [this, reply, file, progress, info, localPath] {
272 file->write(reply->readAll());
273 file->close();
274 progress->close();
275 progress->deleteLater();
276
277 const auto err = reply->error();
278 const qint64 onDisk = QFileInfo(localPath).size();
279 reply->deleteLater();
280 delete file;
281
282 if (err == QNetworkReply::OperationCanceledError) {
283 QFile::remove(localPath);
284 return;
285 }
286 if (err != QNetworkReply::NoError) {
287 QFile::remove(localPath);
288 reportError(tr("Download failed: %1").arg(reply->errorString()));
289 return;
290 }
291 if (info.assetSize > 0 && onDisk != info.assetSize) {
292 QFile::remove(localPath);
293 reportError(tr("Download size mismatch (got %1 bytes, expected %2).")
294 .arg(onDisk).arg(info.assetSize));
295 return;
296 }
297 launchInstaller(localPath);
298 });
299}

301void Updater::launchInstaller(const QString &localPath) {
302#if defined(Q_OS_MACOS)
303 // Mount the DMG in Finder; user drags LASSIE.app into Applications.
304 if (!QDesktopServices::openUrl(QUrl::fromLocalFile(localPath))) {
305 reportError(tr("Couldn't open downloaded installer at %1").arg(localPath));
306 return;
307 }
308 QMessageBox::information(uiParent_, tr("Install LASSIE Update"),
309 tr("The disk image has been mounted. Drag <b>LASSIE</b> into your "
310 "Applications folder, then relaunch."));
311
312#elif defined(Q_OS_WIN)
313 // Launch the NSIS installer; it'll elevate, uninstall the current
314 // version, and install the new one. We quit so it can replace LASSIE.exe.
315 if (!QDesktopServices::openUrl(QUrl::fromLocalFile(localPath))) {
316 reportError(tr("Couldn't launch installer at %1").arg(localPath));
317 return;
318 }
319 QMessageBox::information(uiParent_, tr("Install LASSIE Update"),
320 tr("The installer has been launched. LASSIE will now quit so the "
321 "installer can replace it."));
322 QApplication::quit();
323
324#else
325 // Linux (and unknown OSes): no automatic AppImage swap — too many
326 // failure modes (PID race, alternate install locations, mounted
327 // read-only filesystems). Show the file in a file manager and let
328 // the user move it over their existing AppImage.
329 QDesktopServices::openUrl(QUrl::fromLocalFile(QFileInfo(localPath).absolutePath()));
330 QMessageBox::information(uiParent_, tr("Update Downloaded"),
331 tr("Saved to:\n\n%1\n\nReplace your existing AppImage with this "
332 "file and relaunch.").arg(localPath));
333#endif
334}
