8void writeDomNode(QXmlStreamWriter& writer, const QDomNode& node)
9{
10 if (node.isElement()) {
11 const QDomElement element = node.toElement();
12 writer.writeStartElement(element.tagName());
13 const QDomNamedNodeMap attributes = element.attributes();
14 for (int i = 0; i < attributes.count(); ++i) {
15 const QDomAttr attribute = attributes.item(i).toAttr();
16 writer.writeAttribute(attribute.name(), attribute.value());
17 }
18 for (QDomNode child = element.firstChild(); !child.isNull();
19 child = child.nextSibling()) {
20 writeDomNode(writer, child);
21 }
22 writer.writeEndElement();
23 } else if (node.isCDATASection()) {
24 writer.writeCDATA(node.toCDATASection().data());
25 } else if (node.isText()) {
26 const QString text = node.toText().data();
27 if (!text.trimmed().isEmpty())
28 writer.writeCharacters(text);
29 }
30}

34 // Columns: probability, amplitude, rate, width, spread, direction,

41 /* AMPTRANS */ { true, true, true, true, false, false, false },
42 /* FREQTRANS */ { true, true, true, true, false, false, false },
43 /* WAVE_TYPE */ { false, true, false, false, false, false, false },
44 /* PHASE_MOD */ { true, true, true, false, false, false, false },
45 };
46
47 if (modifier.type >= 8 || fieldIndex < 0 || fieldIndex >= 7
48 || !fields[modifier.type][fieldIndex]) {
