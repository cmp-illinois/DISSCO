113void EnvLibDrawingArea::setActiveNodeCoordinate(const QString& _x, const QString& _y)
114{
115 // Update the selected node's x/y from user text fields
116 EnvelopeLibraryEntry* env = envelopeLibraryWindow->getActiveEnvelope();
117 if (!env || !activeNode) return;
118
119 double newX = _x.toDouble();
120 double newY = _y.toDouble();
121
122 // Apply boundary constraints
123 if (activeNode->leftSeg && activeNode->rightSeg) {
124 // Middle nodes: constrain X between adjacent nodes
125 double leftBound = activeNode->leftSeg->leftNode->x + 0.001;
126 double rightBound = activeNode->rightSeg->rightNode->x - 0.001;
127 newX = qBound(leftBound, newX, rightBound);
128 } else {
129 // Head/tail nodes: only constrain Y
130 newX = activeNode->x; // Keep X unchanged for head/tail
131 }
132
133 // Constrain Y to [0, 4]
134 newY = qBound(0.0, newY, 4.0);
135
136 activeNode->x = newX;
137 activeNode->y = newY;
138
139 MUtilities::modified();
140 adjustBoundary(env);
141 showGraph(env);
142}

176void EnvLibDrawingArea::paintEvent(QPaintEvent* event)
177{
178 Q_UNUSED(event);
179 QPainter painter(this);
180 painter.fillRect(rect(), Qt::white);
181
182 EnvelopeLibraryEntry* env = envelopeLibraryWindow->getActiveEnvelope();
183 if (!env) return;
184
185 int w = width(), h = height();
186
187 // Draw grid lines
188 painter.save();
189 QPen gridPen(QColor(220,220,220));
190 gridPen.setStyle(Qt::DashLine);
191 painter.setPen(gridPen);
192 // Y-axis: 4 lines at 1.0, 2.0, 3.0 (each = 25% of height)
193 for (int i = 1; i < 4; ++i) {
194 int y = h - i * h / 4;
195 painter.drawLine(0, y, w, y);
196 }
197 // X-axis: lines every 0.2 (5 intervals)
198 for (int i = 1; i < 5; ++i) {
199 int x = i * w / 5;
200 painter.drawLine(x, 0, x, h);
201 }
202 painter.restore();
203
204 // Draw axis labels
205 QFont font = painter.font();
206 font.setPointSize(10);
207 painter.setFont(font);
208 painter.setPen(Qt::black);
209 // Y-axis labels along left edge at each grid line (skip 0, shown by X-axis)
210 for (int i = 1; i <= 4; ++i) {
211 int y = h - i * h / 4;
212 painter.drawText(3, y - 3, QString::number(i));
213 }
214 // X-axis labels along bottom at each grid line
215 for (int i = 0; i <= 5; ++i) {
216 int x = i * w / 5;
217 QString label = QString::number(i * 0.2, 'f', 1);
218 painter.drawText(x + 2, h - 3, label);
219 }
220
221 adjustBoundary(env);
222
223 // Draw each segment
224 EnvLibEntrySeg* seg = env->head ? env->head->rightSeg : nullptr;
225 while (seg) {
226 EnvLibEntryNode* L = seg->leftNode;
227 EnvLibEntryNode* R = seg->rightNode;
228 double x1 = L->x * w*w / double(w+1);
229 double y1 = h - getAdjustedY(L->y)*h*h / double(h+1);
230 double x2 = R->x * w*w / double(w+1);
231 double y2 = h - getAdjustedY(R->y)*h*h / double(h+1);
232
233 QPen pen;
234 // color by type (use dark/contrasting colors)
235 if (seg->segmentType == envSegmentTypeLinear)
236 pen.setColor(Qt::blue); // Blue for linear
237 else if (seg->segmentType == envSegmentTypeExponential)
238 pen.setColor(Qt::red); // Red for exponential
239 else
240 pen.setColor(Qt::darkGreen); // Dark green for spline
241
242 pen.setWidth(seg->segmentProperty == envSegmentPropertyFixed ? 1 : 3);
243 painter.setPen(pen);
244
245 // Draw exponential curve for exponential segments
246 if (seg->segmentType == envSegmentTypeExponential) {
247 // Ensure x1 < x2 for the exponential calculation
248 double startx = x1, starty = y1, endx = x2, endy = y2;
249 if (startx > endx) {
250 std::swap(startx, endx);
251 std::swap(starty, endy);
252 }
253
254 // Draw exponential curve using small line segments
255 const double step = 2.0; // Step size in pixels
256 double tempx = startx;
257
258 // Use the exact LASS exponential formula:
259 // value = y1 + (y2 - y1) * ((1 - pow(base, (I * alpha))) / (1 - pow(base, alpha)))
260 const double base = 2.718282; // e
261 double alpha = 3.0;
262 if (starty > endy) {
263 alpha = -alpha; // Reverse curve direction
264 }
265
266 while (tempx < endx - step) {
267 double I = (tempx - startx) / (endx - startx); // normalized time [0,1]
268 double exp_term = (1.0 - pow(base, I * alpha)) / (1.0 - pow(base, alpha));
269 double tempy = starty + (endy - starty) * exp_term;
270
271 double nextx = tempx + step;
272 double nextI = (nextx - startx) / (endx - startx);
273 double next_exp_term = (1.0 - pow(base, nextI * alpha)) / (1.0 - pow(base, alpha));
274 double nexty = starty + (endy - starty) * next_exp_term;
275
276 painter.drawLine(QPointF(tempx, tempy), QPointF(nextx, nexty));
277 tempx = nextx;
278 }
279
280 // Draw final line to endpoint
281 double finalI = (tempx - startx) / (endx - startx);
282 double final_exp_term = (1.0 - pow(base, finalI * alpha)) / (1.0 - pow(base, alpha));
283 painter.drawLine(QPointF(tempx, starty + (endy - starty) * final_exp_term),
284 QPointF(endx, endy));
285 } else {
286 // Draw straight line for linear and spline segments
287 painter.drawLine(QPointF(x1,y1), QPointF(x2,y2));
288 }
289
290 // draw endpoints
291 painter.setBrush(Qt::black); // Black for node points
292 painter.setPen(Qt::NoPen);
293 painter.drawEllipse(QPointF(x1,y1), 5,5);
294 painter.drawEllipse(QPointF(x2,y2), 5,5);
295
296 seg = R->rightSeg;
297 }
298
299 // highlight active node
300 if (activeNode) {
301 double ax = activeNode->x * w*w / double(w+1);
302 double ay = h - getAdjustedY(activeNode->y)*h*h / double(h+1);
303 painter.setBrush(Qt::magenta); // Magenta for active node
304 painter.drawEllipse(QPointF(ax,ay), 8,8);
305 }
306}

338void EnvLibDrawingArea::mousePressEvent(QMouseEvent* event)
339{
340 EnvelopeLibraryEntry* env = envelopeLibraryWindow->getActiveEnvelope();
341 if (!env) { QWidget::mousePressEvent(event); return; }
342
343 activeSegment = nullptr;
344 mouseX = event->x();
345 mouseY = height() - event->y();
346
347 // pick a node within ±5px
348 EnvLibEntryNode* cand = env->head;
349 int w = width(), h = height();
350 bool found = false;
351 while (cand) {
352 double nx = cand->x*w*w/double(w+1);
353 double ny = getAdjustedY(cand->y)*h*h/double(h+1);
354 if (mouseX >= nx-5 && mouseX <= nx+5 &&
355 mouseY >= ny-5 && mouseY <= ny+5) {
356 activeNode = cand;
357 found = true;
358 break;
359 }
360 if (!cand->rightSeg) break;
361 cand = cand->rightSeg->rightNode;
362 }
363 if (!found) {
364 activeNode = nullptr;
365 envelopeLibraryWindow->setEntries("","");
366 } else {
367 envelopeLibraryWindow->setEntries(
368 QString::number(activeNode->x,'f',4),
369 QString::number(activeNode->y,'f',4));
370 }
371 adjustBoundary(env);
372 showGraph(env);
373
374 // right-click → context menu
375 if (event->button() == Qt::RightButton) {
376 actionRemove->setEnabled(activeNode && activeNode->leftSeg && activeNode->rightSeg);
377 m_pMenuPopup->exec(event->globalPos());
378 }
379 // left-click → start drag
380 else if (event->button() == Qt::LeftButton) {
381 mouseLeftButtonPressedDown = true;
382 }
383 QWidget::mousePressEvent(event);
384}

413void EnvLibDrawingArea::insertEnvelopeSegment()
414{
415 EnvelopeLibraryEntry* env = envelopeLibraryWindow->getActiveEnvelope();
416 if (!env) return;
417 MUtilities::modified();
418
419 int w = width(), h = height();
420 double ix = mouseX*(w+1)/double(w*w);
421 double iy = 1.0 - mouseY*(h+1)/double(h*h);
422 iy = mouseAdjustY(iy); // Convert to actual Y value
423 ix = qRound(ix*1000)/1000.0;
424 iy = qRound(iy*1000)/1000.0;
425
426 EnvLibEntryNode* L = env->head;
427 EnvLibEntryNode* R = L->rightSeg->rightNode;
428 while (L->x < ix && R->x < ix) {
429 L = R;
430 R = R->rightSeg->rightNode;
431 }
432
433 // splice in new node & segment
434 EnvLibEntryNode* newN = new EnvLibEntryNode(ix, iy);
435 EnvLibEntrySeg* newS = new EnvLibEntrySeg();
436 L->rightSeg->rightNode = newN;
437 newN->leftSeg = L->rightSeg;
438 newN->rightSeg = newS;
439 newS->leftNode = newN;
440 newS->rightNode = R;
441 R->leftSeg = newS;
442 newS->segmentType = L->rightSeg->segmentType;
443 newS->segmentProperty = L->rightSeg->segmentProperty;
444
445 adjustBoundary(env);
446 showGraph(env);
447}
