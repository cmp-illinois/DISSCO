147 const QString method = nextChildInner(reader);
148 const bool wantRange = (method != QStringLiteral("FUNCTIONS"));
149 // Avoid the togggled-handler's clearNodes (no nodes exist yet

163 if (!node) { reader.skipCurrentElement(); break; }
164 if (wantRange) {
