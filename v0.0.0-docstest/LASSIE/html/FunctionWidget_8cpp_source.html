5QString FunctionWidget::readInner(QXmlStreamReader& r) {
6 // Captures the inner content of the element at the current
7 // StartElement. For text-only elements returns the text; for elements
8 // with nested children returns the serialized inner XML (e.g.
9 // "<Fun>...</Fun>" for an embedded function). On return the reader
10 // is positioned at the matching EndElement.
11 QString result;
12 QXmlStreamWriter w(&result);
13 bool hasChildElement = false;
14 QString textBuffer;
15 int depth = 0;
16 while (!r.atEnd()) {
17 r.readNext();
18 switch (r.tokenType()) {
19 case QXmlStreamReader::StartElement: {
20 hasChildElement = true;
21 w.writeStartElement(r.name().toString());
22 const auto attrs = r.attributes();
23 for (const auto& a : attrs)
24 w.writeAttribute(a.name().toString(), a.value().toString());
25 ++depth;
26 break;
27 }
28 case QXmlStreamReader::EndElement:
29 if (depth == 0)
30 return hasChildElement ? result : textBuffer;
31 w.writeEndElement();
32 --depth;
33 break;
34 case QXmlStreamReader::Characters:
35 if (hasChildElement) w.writeCharacters(r.text().toString());
36 else textBuffer += r.text().toString();
37 break;
38 default: break;
39 }
40 }
41 return hasChildElement ? result : textBuffer;
42}
