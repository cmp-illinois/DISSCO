71 for (int li = 0; li < h.event_layers.size(); ++li) {
72 const QList<Package>& pkgs = h.event_layers[li].discrete_packages;
73 QList<int> hits;

119inline void applyRename(const Assoc& refs, const QString& newName) {
120 for (const auto& entry : refs) {
121 HEvent* h = resolveOwner(entry.first);
122 if (!h || entry.first.layerIndex >= h->event_layers.size()) continue;
123 QList<Package>& pkgs = h->event_layers[entry.first.layerIndex].discrete_packages;

135inline void applyDelete(const Assoc& refs) {
