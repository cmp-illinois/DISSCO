69 size_t CalculateTupletLimit() {
70 size_t tuplet_num = 1;
71
72 while (beat_edus_ % tuplet_num == 0) {
73 valid_dividers_.push_back(beat_edus_ / tuplet_num);
74 tuplet_num++;
75 }

87 int DetermineTuplet(int dur) {
88 for (int tuplet_num = 2; tuplet_num < tuplet_limit_; ++tuplet_num) {
89 int tuplet_beat_dur = beat_edus_ / tuplet_num;
90 if (dur % tuplet_beat_dur == 0) {
91 return tuplet_num;
92 }
93 }

95 return -1; // invalid tuplet

157 static int DiscreteLog2(int num) {
158 if (num % 2 != 0) {
159 return -1;
160 }
161
162 int pow = 0;
163 int power_of_2 = 1;
164 while (power_of_2 < num) { // && pow < std::numeric_limits<int>::digits
165 power_of_2 *= 2;
166 ++pow;
167 if (power_of_2 == num) {
168 return pow;
169 }
170 }
171
172 return -1;
173 }

182 static int CalculateNearestPow2(int num) {
