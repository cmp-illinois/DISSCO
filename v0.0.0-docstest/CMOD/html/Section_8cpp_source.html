377void Section::AddRestsAndFlatten() {
378 for(size_t i = 0; i < section_.size(); ++i) {
379 vector<Note*>::iterator it;
380 Note* prev = *(section_[i].begin());
381 section_flat_.push_back(prev);
382 Note* cur;
383
384 for (it = section_[i].begin() + 1; it != section_[i].end(); it++) {
385 cur = *it;
386 int gap = cur->start_t - prev->end_t;
387
388 // a rest will be placed only if gap is more than half of the
389 // smallest non-zero valid duration
390 if(gap > (time_signature_.beat_edus_ / (time_signature_.tuplet_limit_ - 1) / 2)) {
391 Note* rest = new Note();
392 rest->start_t = prev->end_t;
393 rest->end_t = cur->start_t;
394 rest->pitch_out = "r";
395 rest->type = NoteType::kRest;
396 section_flat_.push_back(rest);
397 } else {
398 if(it+1 == section_[i].end()){
399 prev->end_t = cur->start_t; // Can't shorten current note because end of bar
400 } else {
401 cur->start_t = prev->end_t;
402 }
403 }
404 section_flat_.push_back(cur);
405 prev = cur;
406 }
407 }

410void Section::Notate() {
411 list<Note*>::iterator it = section_flat_.begin();
412 list<Note*>::iterator next_it = ++section_flat_.begin();
413 int prev_tuplet = 0; // the previous tuplet type
414 int tuplet_dur = 0; // the current tuplet duration in edus
415 for (; it != section_flat_.end(); ++it, ++next_it) {
416 Note* cur = *it;
417
418 // adjust note duration according to the tuplet type
419 if(tuplet_dur > 0) {
420 int duration = cur->end_t - cur->start_t;
421 int dur_remainder = (duration) % time_signature_.beat_edus_;
422 int dur_beats = (duration) / time_signature_.beat_edus_;
423 int desire_type = prev_tuplet;
424 int cur_type = time_signature_.DetermineTuplet(dur_remainder);
425 int excess_tuplet_type = time_signature_.DetermineTuplet(duration - tuplet_dur);
426
427 // True if the note exceeds the current tuplet duration
428 // but the excess is expressible in another tuplet
429 bool note_is_valid = (duration > tuplet_dur) && (excess_tuplet_type != -1);
430
431 // if the current note's duration cannot be fitted in the tuplet
432 // find the closest value and change the end time of the note.
433 // Also change the start time of the next note
434 if ((desire_type != cur_type) && !note_is_valid) {
435 int t = time_signature_.beat_edus_ / desire_type;
436 double a = (double)dur_remainder / (double)t;
437 int best_fit = (int) round(a) * t;
438
439 cur->end_t = cur->start_t + dur_beats * time_signature_.beat_edus_ + best_fit;
440
441 Note* next_note = *(next_it);
442 if (next_note == 0 || next_it == section_flat_.end()) {
443 continue;
444 }
445
446 next_note->start_t = cur->start_t + dur_beats * time_signature_.beat_edus_ + best_fit;
447 }
448 }
449
450 // force the closing of the tuplet before the bar line (not necessary if
451 // the code is correct)
452 if(cur->type_out == "\\bar\"|\" \n" || cur->type_out == " "){
453 if(tuplet_dur > 0){
454 cur->type_out = "} " + cur->type_out;
455 tuplet_dur = 0;
456 }
457 continue;
458 }
459
460 tuplet_dur = NotateCurrentNote(cur, &prev_tuplet, tuplet_dur);
461 }

464void Section::CapEnding() {
465 int cur_bar_edus = 0;
466 list<Note*> last_bar = PopLastBarNotes();
467
468 if (!last_bar.empty()) {
469 cur_bar_edus = last_bar.back()->end_t - last_bar.front()->start_t;
470 cur_bar_edus %= time_signature_.bar_edus_; // want the last __incomplete__ bar
471 }
472
473 int used_edus_ = last_bar.back()->end_t; // edus start at 0, so this gives used time
474 remaining_edus_ -= used_edus_; // 'rest' edus
475 int total_edus_to_use = remaining_edus_ + cur_bar_edus;
476
477 if (remaining_edus_ < 0) {
478 cerr << "Sections overlap" << endl;
479 exit(1);
480 } else if (remaining_edus_ == 0 && cur_bar_edus == 0) {
481 return; // Sections align perfectly!
482 } else {
483 int pow_2 = 0;
484 int min_err = INT_MAX;
485 int ts_num, ts_den;
486 while (time_signature_.beat_edus_ % TimeSignature::Power(2, pow_2) == 0) {
487 int tmp_beat_edus = time_signature_.beat_edus_ / TimeSignature::Power(2, pow_2);
488 if (total_edus_to_use % tmp_beat_edus == 0) {
489 ts_num = total_edus_to_use / tmp_beat_edus;
490 ts_den = time_signature_.unit_note_ * TimeSignature::Power(2, pow_2);
491 min_err = 0;
492 break; // Overhanging time forms a dyadic time signature
493 } else {
494 // Form a dyadic time signature by adding sound or rest with the least error
495 int num_beats = (total_edus_to_use / tmp_beat_edus) + 1; // TODO - do this better
496 int err = tmp_beat_edus * num_beats - total_edus_to_use;
497 if (err < min_err) {
498 ts_num = num_beats; // Add time
499 ts_den = time_signature_.unit_note_ * TimeSignature::Power(2, pow_2);
500 min_err = err;
501 }
502 }
503 ++pow_2;
504 }
505
506 int beat_divisor = TimeSignature::Power(2, pow_2);
507 Tempo new_tempo(time_signature_.tempo_);
508
509 new_tempo.setEDUPerTimeSignatureBeat(time_signature_.beat_edus_ / beat_divisor);
510 new_tempo.setTimeSignature(Note::int_to_str(ts_num) + "/" + Note::int_to_str(ts_den));
511 // IMPORTANT - Tempo rate (i.e. 1/4=60bpm) is left the same
512 // IMPORTANT - Tempo start time left unchanged because the calculation is not necessary
513
514 cap_ = new Section(TimeSignature(new_tempo));
515 cap_->SetDurationEDUS(-1);
516
517 int offset = last_bar.front()->start_t;
518 while (!last_bar.empty()) {
519 last_bar.front()->start_t -= offset; // make notes in cap start from 0
520 last_bar.front()->end_t -= offset;
521
522 if (!cap_->InsertNote(last_bar.front())) {
523 cerr << "Note could not be inserted into end cap. " <<
524 "This should not happen under any circumstance." << endl;
525 exit(1);
526 }
527 last_bar.pop_front();
528 }
529
530 if (min_err != 0 && remaining_edus_ == 0) { // No extra time and leftover sound does not fill time signature
531 Note* extra_space = new Note(*last_bar.back()); // TODO - what if we get a tie over the last bar
532 extra_space->start_t = last_bar.back()->start_t;
533 extra_space->end_t = extra_space->start_t + min_err;
534 extra_space->split = 1;
535 cap_->InsertNote(extra_space);
536 }
537
538 if (min_err != 0) {
539 cout << Note::int_to_str(new_tempo.calculateSecondsFromEDUs(min_err))
540 << " seconds added to stitch sections." << endl;
541 }
542
543 // Only notate time signature if different
544 cap_->time_signature_ == time_signature_ ? cap_->Build(false) : cap_->Build(true);
545
546 list<Note*> final_bar = cap_->PopFirstBar();
547 while (!final_bar.empty()) {
548 section_flat_.push_back(final_bar.front());
549 final_bar.pop_front();
550 }
551 }
552}

554int Section::NotateCurrentNote(Note* current_note,
555 int* prev_tuplet,
556 int tuplet_dur) {
557 int dur = current_note->end_t - current_note->start_t;
558 if (dur == 0) {
559 return tuplet_dur;
560 }
561
562 int remaining_dur = dur;
563 if (tuplet_dur > 0) {
564 remaining_dur = FillCurrentTupletDur(current_note, *prev_tuplet, tuplet_dur);
565 if (remaining_dur < 0) { // Note fits inside the current tuplet
566 return -remaining_dur; // Tuplet partially filled
567 }
568 tuplet_dur = 0; // Tuplet now filled
569 }
570
571 remaining_dur = FillCompleteBeats(current_note, remaining_dur);
572
573 if (remaining_dur > 0) {
574 tuplet_dur = CreateTupletWithRests(current_note, prev_tuplet, remaining_dur);
575 }
576
577 return tuplet_dur;
578}

580int Section::FillCurrentTupletDur(Note* current_note,
581 int prev_tuplet,
582 int tuplet_dur) {
583 int dur = current_note->end_t - current_note->start_t;
584 // if the current duration is less than the duration of the tuplet,
585 // the entire duration will be inserted in the tuplet and the tuplet will
586 // be completed by the next sound or silence.
587 if (dur < tuplet_dur) {
588 NoteInTuplet(current_note, prev_tuplet, dur);
589 return dur - tuplet_dur;
590 }
591 // even if the previous tuplet is an eighth note or sixteenth note,
592 // it is still necessary to split part of the current note.
593 if (prev_tuplet == 2 || prev_tuplet == 4) {
594 int unit = tuplet_dur / (time_signature_.beat_edus_ / prev_tuplet);
595 if(unit == 3) {
596 string s = Note::int_to_str(time_signature_.unit_note_ * 2);
597 current_note->type_out += current_note->pitch_out + s + ".";
598 LoudnessMark(current_note); // this places all the dynamic marks at the beginning of a sound
599 ModifiersMark(current_note);
600 } else {
601 string s = Note::int_to_str(time_signature_.unit_note_ * prev_tuplet / unit);
602 current_note->type_out += current_note->pitch_out + s;
603 LoudnessMark(current_note); // this places all the dynamic marks at the beginning of a sound
604 ModifiersMark(current_note);
605 }
606 if ((dur > tuplet_dur || current_note->split == 1) &&
607 (current_note->pitch_out != "r")) {
608 current_note->type_out += "~ ";
609 current_note->split = 0;
610 } else {
611 current_note->type_out += " ";
612 }
613
614 } else {
615 // if the current sound completes the tuplet use the LilyPond symbol and
616 // close the tuplet
617 NoteInTuplet(current_note, prev_tuplet, tuplet_dur);
618 if ((dur > tuplet_dur || current_note->split == 1) && (current_note->pitch_out != "r")) {
619 current_note->type_out += "~ ";
620 current_note->split = 0;
621 } else {
622 // current_note->loudness_and_modifiers();
623 }
624 current_note->type_out += "} ";
625 }
626
627 return dur - tuplet_dur;
628}

630int Section::FillCompleteBeats(Note* current_note, int remaining_dur) {
631 int remainder = remaining_dur % time_signature_.beat_edus_;
632 int mainDur = remaining_dur / time_signature_.beat_edus_;
