120// Element-only navigation helpers for pugi::xml_node. pugixml's first_child()
121// and next_sibling() return any node type (including whitespace), so wrap them
122// to skip non-element nodes, matching Xerces' getFirstElementChild semantics.
123inline pugi::xml_node GFEC(pugi::xml_node n) {
124 auto c = n.first_child();
125 while (c && c.type() != pugi::node_element) c = c.next_sibling();
126 return c;
127}
128inline pugi::xml_node GNES(pugi::xml_node n) {
129 auto c = n.next_sibling();
130 while (c && c.type() != pugi::node_element) c = c.next_sibling();
131 return c;
132}

134// Recursive descendant lookup by tag name (Xerces getElementsByTagName first hit).
135inline pugi::xml_node descendantByName(pugi::xml_node root, const char* tag) {
136 if (!root) return pugi::xml_node();
137 for (auto c = root.first_child(); c; c = c.next_sibling()) {
138 if (c.type() == pugi::node_element && std::strcmp(c.name(), tag) == 0) return c;
139 auto found = descendantByName(c, tag);
140 if (found) return found;
141 }
142 return pugi::xml_node();
143}
