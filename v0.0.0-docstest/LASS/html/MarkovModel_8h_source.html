142template<typename T>
143void MarkovModel<T>::makeConsistent() {
144 for (unsigned i = 0; i < transitionMatrix.size(); i++) {
145 vector<double>& row = transitionMatrix[i];
146 // find row sum
147 double sum = 0.0;
148 for (unsigned i = 0; i < row.size(); i++) {
149 sum += row[i];
150 }
151
152 if (sum == 0) { // when sum is 0, use uniform distribution
153 for (unsigned i = 0; i < row.size(); i++) {
154 row[i] = 1.0 / row.size();
155 }
156 } else { // otherwise, normalize
157 for (unsigned i = 0; i < row.size(); i++) {
158 row[i] /= sum;
159 }
160 }
161 }
162
163 // make sure initial distribution is a distribution
164 double sum = 0.0;
165 for (unsigned i = 0; i < initialDistribution.size(); i++) {
166 sum += initialDistribution[i];
167 }
168 if (sum == 0) {
169 for (unsigned i = 0; i < initialDistribution.size(); i++) {
170 initialDistribution[i] = 1.0 / initialDistribution.size();
171 }
172 } else {
173 for (unsigned i = 0; i < initialDistribution.size(); i++) {
174 initialDistribution[i] /= sum;
175 }
176 }
177}
