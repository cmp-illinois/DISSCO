43LPCombFilter::LPCombFilter(float gain, long delay, float lpf_gain)
44{
45 // set parameters
46 g = gain;
47 D = delay;
48 lpf_g = lpf_gain;
49
50 // create the lowpass-feedback filter
51 lpf = new LowPassFilter(lpf_g);
52
53 // initialize x_hist queue
54 x_hist = new Filter::hist_queue<m_sample_type>(D);
55 for(long i=0; i<D; i++)
56 x_hist->enqueue(0.0);
57}
58//----------------------------------------------------------------------------//
59LPCombFilter::~LPCombFilter()
60{
61 delete lpf;
62 delete x_hist;
63}

65//----------------------------------------------------------------------------//
66m_sample_type LPCombFilter::do_filter(m_sample_type x_t)
67{
68 m_sample_type y_t;
69
70 y_t = x_hist->dequeue();
71 x_hist->enqueue(x_t + (g * lpf->do_filter(y_t)));
72
73 return y_t;
74}
