116void SoundSample::composite(SoundSample& ss, m_time_type startTime)
117{
118 // make sure both sound samples have the same sampling rate.
119 if (samplingRate_ != ss.samplingRate_)
120 {
121 cerr << "ERROR: SoundSample::composite() "
122 << "two SoundSamples have different sampling rates. "
123 << "Bailing out." << endl;
124 return;
125 }
126
127 // find the number of samples to composite
128 m_sample_count_type samplesToCopy = ss.data_.size();
129 m_sample_count_type samplesToSkip =
130 m_sample_count_type(startTime * float(samplingRate_));
131 m_sample_count_type lengthNeeded = samplesToCopy + samplesToSkip;
132
133 if ( m_sample_count_type(data_.size()) < lengthNeeded)
134 {
135 if ( m_sample_count_type(data_.size()) + MIN_CLIP_WARNING < lengthNeeded)
136 {
137 cerr << "WARNING: SoundSample::composite() "
138 << "sample being composited past end: "
139 << "clipping will occur." << endl;
140 }
141
142 samplesToCopy =
143 m_sample_count_type(data_.size()) - samplesToSkip;
144
145 if (samplesToCopy <= 0) return;
146 }
147
148 // copy the data:
149 for (m_sample_count_type s=0; s<samplesToCopy; s++)
150 {
151 data_[s+samplesToSkip] += ss.data_[s];
152 }
153}
